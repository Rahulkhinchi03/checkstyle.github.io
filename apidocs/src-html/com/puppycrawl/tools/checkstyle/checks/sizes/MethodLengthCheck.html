020package com.puppycrawl.tools.checkstyle.checks.sizes;
021
022import java.util.ArrayDeque;
023import java.util.Deque;
024import java.util.HashSet;
025import java.util.Objects;
026import java.util.Set;
027import java.util.stream.IntStream;
028import java.util.stream.Stream;
029
030import com.puppycrawl.tools.checkstyle.StatelessCheck;
031import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
032import com.puppycrawl.tools.checkstyle.api.DetailAST;
033import com.puppycrawl.tools.checkstyle.api.TokenTypes;
034import com.puppycrawl.tools.checkstyle.utils.CommonUtil;
035
036/**
037 * <p>
038 * Checks for long methods and constructors.
039 * </p>
040 * <p>
041 * Rationale: If a method becomes very long it is hard to understand.
042 * Therefore long methods should usually be refactored into several
043 * individual methods that focus on a specific task.
044 * </p>
045 * <ul>
046 * <li>
047 * Property {@code max} - Specify the maximum number of lines allowed.
048 * Type is {@code int}.
049 * Default value is {@code 150}.
050 * </li>
051 * <li>
052 * Property {@code countEmpty} - Control whether to count empty lines and comments.
053 * Type is {@code boolean}.
054 * Default value is {@code true}.
055 * </li>
056 * <li>
057 * Property {@code tokens} - tokens to check
058 * Type is {@code java.lang.String[]}.
059 * Validation type is {@code tokenSet}.
060 * Default value is:
061 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
062 * METHOD_DEF</a>,
063 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
064 * CTOR_DEF</a>,
065 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#COMPACT_CTOR_DEF">
066 * COMPACT_CTOR_DEF</a>.
067 * </li>
068 * </ul>
069 * <p>
070 * To configure the check:
071 * </p>
072 * <pre>
073 * &lt;module name="MethodLength"/&gt;
074 * </pre>
075 * <p>
076 * Example:
077 * </p>
078 * <pre>
079 * public class MyClass {
080 *   public MyClass() {  // constructor (line 1)
081 *        /&#42; line 2
082 *            ...
083 *           line 150 &#42;/
084 *   } // line 151, violation, as it is over 150
085 *
086 *   public void firstExample() { // line 1
087 *
088 *       // line 3
089 *       System.out.println("line 4");
090 *       /&#42; line 5
091 *          line 6 &#42;/
092 *   } // line 7, OK, as it is less than 150
093 *
094 *   public void secondExample() { // line 1
095 *       // line 2
096 *       System.out.println("line 3");
097 *
098 *       /&#42; line 5
099 *           ...
100 *          line 150 &#42;/
101 *   } // line 151, violation, as it is over 150
102 * }
103 * </pre>
104 * <p>
105 * To configure the check so that it accepts methods with at most 4 lines:
106 * </p>
107 * <pre>
108 * &lt;module name="MethodLength"&gt;
109 *   &lt;property name="tokens" value="METHOD_DEF"/&gt;
110 *   &lt;property name="max" value="4"/&gt;
111 * &lt;/module&gt;
112 * </pre>
113 * <p>
114 * Example:
115 * </p>
116 * <pre>
117 * public class MyTest {
118 *   public MyTest()  {            // constructor (line 1)
119 *       int var1 = 2;             // line 2
120 *       int var2 = 4;             // line 3
121 *       int sum = var1 + var2;  // line 4
122 *   } // line 5, OK, constructor is not mentioned in the tokens
123 *
124 *   public void firstMethod() { // line 1
125 *       // comment (line 2)
126 *       System.out.println("line 3");
127 *   } // line 4, OK, as it allows at most 4 lines
128 *
129 *   public void secondMethod() { // line 1
130 *       int index = 0;   // line 2
131 *       if (index &#60; 5) { // line 3
132 *           index++;     // line 4
133 *       }                // line 5
134 *   } // line 6, violation, as it is over 4 lines
135 *
136 *   public void thirdMethod() { // line 1
137 *
138 *       // comment (line 3)
139 *       System.out.println("line 4");
140 *   } // line 5, violation, as it is over 4 lines
141 * }
142 * </pre>
143 * <p>
144 * To configure the check so that it accepts methods with at most 4 lines,
145 * not counting empty lines and comments:
146 * </p>
147 * <pre>
148 * &lt;module name="MethodLength"&gt;
149 *   &lt;property name="tokens" value="METHOD_DEF"/&gt;
150 *   &lt;property name="max" value="4"/&gt;
151 *   &lt;property name="countEmpty" value="false"/&gt;
152 * &lt;/module&gt;
153 * </pre>
154 * <p>
155 * Example:
156 * </p>
157 * <pre>
158 * public class MyTest {
159 *   public MyTest()  {          // constructor (line 1)
160 *       int var1 = 2;           // line 2
161 *       int var2 = 4;           // line 3
162 *       int sum = var1 + var2;  // line 4
163 *   } // line 5, OK, constructor is not mentioned in the tokens
164 *
165 *   public void firstMethod() { // line 1
166 *       // comment - not counted as line
167 *       System.out.println("line 2");
168 *   } // line 3, OK, as it allows at most 4 lines
169 *
170 *   public void secondMethod() { // line 1
171 *       int index = 0;   // line 2
172 *       if (index &#60; 5) { // line 3
173 *           index++;     // line 4
174 *       }                // line 5
175 *   } // line 6, violation, as it is over 4 lines
176 *
177 *   public void thirdMethod() { // line 1
178 *
179 *       // comment - not counted as line
180 *       System.out.println("line 2");
181 *   } // line 3, OK, as it allows at most 4 lines
182 * }
183 * </pre>
184 * <p>
185 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
186 * </p>
187 * <p>
188 * Violation Message Keys:
189 * </p>
190 * <ul>
191 * <li>
192 * {@code maxLen.method}
193 * </li>
194 * </ul>
195 *
196 * @since 3.0
197 */
198@StatelessCheck
199public class MethodLengthCheck extends AbstractCheck {
200
201    /**
202     * A key is pointing to the warning message text in "messages.properties"
203     * file.
204     */
205    public static final String MSG_KEY = "maxLen.method";
206
207    /** Default maximum number of lines. */
208    private static final int DEFAULT_MAX_LINES = 150;
209
210    /** Control whether to count empty lines and comments. */
211    private boolean countEmpty = true;
212
213    /** Specify the maximum number of lines allowed. */
214    private int max = DEFAULT_MAX_LINES;
215
216    @Override
217    public int[] getDefaultTokens() {
218        return getAcceptableTokens();
219    }
220
221    @Override
222    public int[] getAcceptableTokens() {
223        return new int[] {
224            TokenTypes.METHOD_DEF,
225            TokenTypes.CTOR_DEF,
226            TokenTypes.COMPACT_CTOR_DEF,
227        };
228    }
229
230    @Override
231    public int[] getRequiredTokens() {
232        return CommonUtil.EMPTY_INT_ARRAY;
233    }
234
235    @Override
236    public void visitToken(DetailAST ast) {
237        final DetailAST openingBrace = ast.findFirstToken(TokenTypes.SLIST);
238        if (openingBrace != null) {
239            final int length;
240            if (countEmpty) {
241                final DetailAST closingBrace = openingBrace.findFirstToken(TokenTypes.RCURLY);
242                length = getLengthOfBlock(openingBrace, closingBrace);
243            }
244            else {
245                length = countUsedLines(openingBrace);
246            }
247            if (length > max) {
248                final String methodName = ast.findFirstToken(TokenTypes.IDENT).getText();
249                log(ast, MSG_KEY, length, max, methodName);
250            }
251        }
252    }
253
254    /**
255     * Returns length of code.
256     *
257     * @param openingBrace block opening brace
258     * @param closingBrace block closing brace
259     * @return number of lines with code for current block
260     */
261    private static int getLengthOfBlock(DetailAST openingBrace, DetailAST closingBrace) {
262        final int startLineNo = openingBrace.getLineNo();
263        final int endLineNo = closingBrace.getLineNo();
264        return endLineNo - startLineNo + 1;
265    }
266
267    /**
268     * Count number of used code lines without comments.
269     *
270     * @param ast start ast
271     * @return number of used lines of code
272     */
273    private static int countUsedLines(DetailAST ast) {
274        final Deque<DetailAST> nodes = new ArrayDeque<>();
275        nodes.add(ast);
276        final Set<Integer> usedLines = new HashSet<>();
277        while (!nodes.isEmpty()) {
278            final DetailAST node = nodes.removeFirst();
279            final int lineNo = node.getLineNo();
280            // text block requires special treatment,
281            // since it is the only non-comment token that can span more than one line
282            if (node.getType() == TokenTypes.TEXT_BLOCK_LITERAL_BEGIN) {
283                IntStream.rangeClosed(lineNo, node.getLastChild().getLineNo())
284                    .forEach(usedLines::add);
285            }
286            else {
287                usedLines.add(lineNo);
288                Stream.iterate(
289                    node.getLastChild(), Objects::nonNull, DetailAST::getPreviousSibling
290                ).forEach(nodes::addFirst);
291            }
292        }
293        return usedLines.size();
294    }
295
296    /**
297     * Setter to specify the maximum number of lines allowed.
298     *
299     * @param length the maximum length of a method.
300     */
301    public void setMax(int length) {
302        max = length;
303    }
304
305    /**
306     * Setter to control whether to count empty lines and comments.
307     *
308     * @param countEmpty whether to count empty and comments.
309     */
310    public void setCountEmpty(boolean countEmpty) {
311        this.countEmpty = countEmpty;
312    }
313
314}
