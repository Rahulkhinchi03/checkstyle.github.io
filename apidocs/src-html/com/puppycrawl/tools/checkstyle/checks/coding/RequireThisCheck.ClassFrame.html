039/**
040 * <p>
041 * Checks that references to instance variables and methods of the present
042 * object are explicitly of the form "this.varName" or "this.methodName(args)"
043 * and that those references don't rely on the default behavior when "this." is absent.
044 * </p>
045 * <p>Warning: the Check is very controversial if 'validateOnlyOverlapping' option is set to 'false'
046 * and not that actual nowadays.</p>
047 * <p>Rationale:</p>
048 * <ol>
049 *   <li>
050 *     The same notation/habit for C++ and Java (C++ have global methods, so having
051 *     &quot;this.&quot; do make sense in it to distinguish call of method of class
052 *     instead of global).
053 *   </li>
054 *   <li>
055 *     Non-IDE development (ease of refactoring, some clearness to distinguish
056 *     static and non-static methods).
057 *   </li>
058 * </ol>
059 * <p>Limitations: Nothing is currently done about static variables
060 * or catch-blocks.  Static methods invoked on a class name seem to be OK;
061 * both the class name and the method name have a DOT parent.
062 * Non-static methods invoked on either this or a variable name seem to be
063 * OK, likewise.
064 * </p>
065 * <ul>
066 * <li>
067 * Property {@code checkFields} - Control whether to check references to fields.
068 * Type is {@code boolean}.
069 * Default value is {@code true}.
070 * </li>
071 * <li>
072 * Property {@code checkMethods} - Control whether to check references to methods.
073 * Type is {@code boolean}.
074 * Default value is {@code true}.
075 * </li>
076 * <li>
077 * Property {@code validateOnlyOverlapping} - Control whether to check only
078 * overlapping by variables or arguments.
079 * Type is {@code boolean}.
080 * Default value is {@code true}.
081 * </li>
082 * </ul>
083 * <p>
084 * To configure the default check:
085 * </p>
086 * <pre>
087 * &lt;module name=&quot;RequireThis&quot;/&gt;
088 * </pre>
089 * <p>Example:</p>
090 * <pre>
091 * public class Test {
092 *   private int a;
093 *   private int b;
094 *   private int c;
095 *
096 *   public Test(int a) {
097 *     // overlapping by constructor argument
098 *     this.a = a;       // OK, this keyword used
099 *     b = 0;            // OK, no overlap
100 *     foo(5);           // OK
101 *   }
102 *
103 *   public void foo(int c) {
104 *     // overlapping by method argument
105 *     c = c;            // violation, reference to instance variable "c" requires "this"
106 *   }
107 * }
108 * </pre>
109 * <p>
110 * To configure the check for fields only:
111 * </p>
112 * <pre>
113 * &lt;module name=&quot;RequireThis&quot;&gt;
114 *   &lt;property name=&quot;checkMethods&quot; value=&quot;false&quot;/&gt;
115 * &lt;/module&gt;
116 * </pre>
117 * <p>Example:</p>
118 * <pre>
119 * public class Test {
120 *   private int a;
121 *   private int b;
122 *   private int c;
123 *
124 *   public Test(int a) {
125 *     // overlapping by constructor argument
126 *     this.a = a;       // OK, this keyword used
127 *     b = 0;            // OK, no overlap
128 *     foo(5);           // OK, no validation for methods
129 *   }
130 *
131 *   public void foo(int c) {
132 *     // overlapping by method argument
133 *     c = c;            // violation, reference to instance variable "c" requires "this"
134 *   }
135 * }
136 * </pre>
137 * <p>
138 * To configure the check for methods only:
139 * </p>
140 * <pre>
141 * &lt;module name=&quot;RequireThis&quot;&gt;
142 *   &lt;property name=&quot;checkFields&quot; value=&quot;false&quot;/&gt;
143 * &lt;/module&gt;
144 * </pre>
145 * <p>Example:</p>
146 * <pre>
147 * public class Test {
148 *   private int a;
149 *   private int b;
150 *   private int c;
151 *
152 *   public Test(int a) {
153 *     // overlapping by constructor argument
154 *     this.a = a;       // OK, no validation for fields
155 *     b = 0;            // OK, no validation for fields
156 *     foo(5);           // OK, no overlap
157 *   }
158 *
159 *   public void foo(int c) {
160 *     // overlapping by method argument
161 *     c = c;            // OK, no validation for fields
162 *   }
163 * }
164 * </pre>
165 * <p>
166 * Note that method call foo(5) does not raise a violation
167 * because methods cannot be overlapped in java.
168 * </p>
169 * <p>
170 * To configure the check to validate for non-overlapping fields and methods:
171 * </p>
172 * <pre>
173 * &lt;module name=&quot;RequireThis&quot;&gt;
174 *   &lt;property name=&quot;validateOnlyOverlapping&quot; value=&quot;false&quot;/&gt;
175 * &lt;/module&gt;
176 * </pre>
177 * <p>Example:</p>
178 * <pre>
179 * public class Test {
180 *   private int a;
181 *   private int b;
182 *   private int c;
183 *
184 *   public Test(int a) {
185 *     // overlapping by constructor argument
186 *     this.a = a;       // OK, no validation for fields
187 *     b = 0;            // violation, reference to instance variable "b" requires "this"
188 *     foo(5);           // violation, method call "foo(5)" requires "this"
189 *   }
190 *
191 *   public void foo(int c) {
192 *     // overlapping by method argument
193 *     c = c;            // violation, reference to instance variable "c" requires "this"
194 *   }
195 * }
196 * </pre>
197 * <p>
198 * Please, be aware of the following logic, which is implemented in the check:
199 * </p>
200 * <p>
201 * 1) If you arrange 'this' in your code on your own, the check will not raise violation for
202 * variables which use 'this' to reference a class field, for example:
203 * </p>
204 * <pre>
205 * public class C {
206 *   private int scale;
207 *   private int x;
208 *
209 *   public void foo(int scale) {
210 *     scale = this.scale;      // no violation
211 *
212 *     if (scale &gt; 0) {
213 *       scale = -scale;        // no violation
214 *     }
215 *     x *= scale;
216 *   }
217 * }
218 * </pre>
219 * <p>
220 * 2) If method parameter is returned from the method, the check will not raise violation for
221 * returned variable/parameter, for example:
222 * </p>
223 * <pre>
224 * public class D {
225 *   private String prefix;
226 *
227 *   public String modifyPrefix(String prefix) {
228 *     prefix = "^" + prefix + "$";  // no violation, because method parameter is returned
229 *     return prefix;
230 *   }
231 * }
232 * </pre>
233 * <p>
234 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
235 * </p>
236 * <p>
237 * Violation Message Keys:
238 * </p>
239 * <ul>
240 * <li>
241 * {@code require.this.method}
242 * </li>
243 * <li>
244 * {@code require.this.variable}
245 * </li>
246 * </ul>
247 *
248 * @since 3.4
249 */
250// -@cs[ClassDataAbstractionCoupling] This check requires to work with and identify many frames.

372    @Override
373    public void beginTree(DetailAST rootAST) {
374        frames = new HashMap<>();
375        current.clear();
376
377        final Deque<AbstractFrame> frameStack = new LinkedList<>();
378        DetailAST curNode = rootAST;
379        while (curNode != null) {
380            collectDeclarations(frameStack, curNode);
381            DetailAST toVisit = curNode.getFirstChild();
382            while (curNode != null && toVisit == null) {
383                endCollectingDeclarations(frameStack, curNode);
384                toVisit = curNode.getNextSibling();
385                if (toVisit == null) {
386                    curNode = curNode.getParent();
387                }
388            }
389            curNode = toVisit;
390        }
391    }

393    @Override
394    public void visitToken(DetailAST ast) {
395        switch (ast.getType()) {
396            case TokenTypes.IDENT:
397                processIdent(ast);
398                break;
399            case TokenTypes.CLASS_DEF:
400            case TokenTypes.INTERFACE_DEF:
401            case TokenTypes.ENUM_DEF:
402            case TokenTypes.ANNOTATION_DEF:
403            case TokenTypes.SLIST:
404            case TokenTypes.METHOD_DEF:
405            case TokenTypes.CTOR_DEF:
406            case TokenTypes.LITERAL_FOR:
407            case TokenTypes.RECORD_DEF:
408                current.push(frames.get(ast));
409                break;
410            default:
411                // do nothing
412        }
413    }

419            case TokenTypes.INTERFACE_DEF:

425            case TokenTypes.LITERAL_FOR:
426            case TokenTypes.RECORD_DEF:
427                current.pop();
428                break;
429            default:
430                // do nothing
431        }
432    }

434    /**
435     * Checks if a given IDENT is method call or field name which
436     * requires explicit {@code this} qualifier.
437     *
438     * @param ast IDENT to check.
439     */
440    private void processIdent(DetailAST ast) {
441        int parentType = ast.getParent().getType();
442        if (parentType == TokenTypes.EXPR
443                && ast.getParent().getParent().getParent().getType()
444                    == TokenTypes.ANNOTATION_FIELD_DEF) {
445            parentType = TokenTypes.ANNOTATION_FIELD_DEF;
446        }
447        switch (parentType) {
448            case TokenTypes.ANNOTATION_MEMBER_VALUE_PAIR:
449            case TokenTypes.ANNOTATION:
450            case TokenTypes.ANNOTATION_FIELD_DEF:
451                // no need to check annotations content
452                break;
453            case TokenTypes.METHOD_CALL:
454                if (checkMethods) {
455                    final AbstractFrame frame = getMethodWithoutThis(ast);
456                    if (frame != null) {
457                        logViolation(MSG_METHOD, ast, frame);
458                    }
459                }
460                break;
461            default:
462                if (checkFields) {
463                    final AbstractFrame frame = getFieldWithoutThis(ast, parentType);
464                    final boolean canUseThis = !isInCompactConstructor(ast);
465                    if (frame != null && canUseThis) {
466                        logViolation(MSG_VARIABLE, ast, frame);
467                    }
468                }
469                break;
470        }
471    }

473    /**
474     * Helper method to log a Violation.
475     *
476     * @param msgKey key to locale message format.
477     * @param ast a node to get line id column numbers associated with the message.
478     * @param frame the class frame where the violation is found.
479     */
480    private void logViolation(String msgKey, DetailAST ast, AbstractFrame frame) {
481        if (frame.getFrameName().equals(getNearestClassFrameName())) {
482            log(ast, msgKey, ast.getText(), "");
483        }
484        else if (!(frame instanceof AnonymousClassFrame)) {
485            log(ast, msgKey, ast.getText(), frame.getFrameName() + '.');
486        }
487    }

489    /**
490     * Returns the frame where the field is declared, if the given field is used without
491     * 'this', and null otherwise.
492     *
493     * @param ast field definition ast token.
494     * @param parentType type of the parent.
495     * @return the frame where the field is declared, if the given field is used without
496     *         'this' and null otherwise.
497     */
498    private AbstractFrame getFieldWithoutThis(DetailAST ast, int parentType) {
499        final boolean importOrPackage = ScopeUtil.getSurroundingScope(ast) == null;
500        final boolean typeName = parentType == TokenTypes.TYPE
501                || parentType == TokenTypes.LITERAL_NEW;
502        AbstractFrame frame = null;
503
504        if (!importOrPackage
505                && !typeName
506                && !isDeclarationToken(parentType)
507                && !isLambdaParameter(ast)) {
508            final AbstractFrame fieldFrame = findClassFrame(ast, false);
509
510            if (fieldFrame != null && ((ClassFrame) fieldFrame).hasInstanceMember(ast)) {
511                frame = getClassFrameWhereViolationIsFound(ast);
512            }
513        }
514        return frame;
515    }

517    /**
518     * Return whether ast is in a COMPACT_CTOR_DEF.
519     *
520     * @param ast The token to check
521     * @return true if ast is in a COMPACT_CTOR_DEF, false otherwise
522     */
523    private static boolean isInCompactConstructor(DetailAST ast) {
524        boolean isInCompactCtor = false;
525        DetailAST parent = ast.getParent();
526        while (parent != null) {
527            if (parent.getType() == TokenTypes.COMPACT_CTOR_DEF) {
528                isInCompactCtor = true;
529                break;
530            }
531            parent = parent.getParent();
532        }
533        return isInCompactCtor;
534    }

536    /**
537     * Parses the next AST for declarations.
538     *
539     * @param frameStack stack containing the FrameTree being built.
540     * @param ast AST to parse.
541     */
542    // -@cs[JavaNCSS] This method is a big switch and is too hard to remove.
543    private static void collectDeclarations(Deque<AbstractFrame> frameStack, DetailAST ast) {
544        final AbstractFrame frame = frameStack.peek();
545        switch (ast.getType()) {
546            case TokenTypes.VARIABLE_DEF:
547                collectVariableDeclarations(ast, frame);
548                break;
549            case TokenTypes.RECORD_COMPONENT_DEF:
550                final DetailAST componentIdent = ast.findFirstToken(TokenTypes.IDENT);
551                ((ClassFrame) frame).addInstanceMember(componentIdent);
552                break;
553            case TokenTypes.PARAMETER_DEF:
554                if (!CheckUtil.isReceiverParameter(ast)
555                        && !isLambdaParameter(ast)
556                        && ast.getParent().getType() != TokenTypes.LITERAL_CATCH) {
557                    final DetailAST parameterIdent = ast.findFirstToken(TokenTypes.IDENT);
558                    frame.addIdent(parameterIdent);
559                }
560                break;
561            case TokenTypes.CLASS_DEF:
562            case TokenTypes.INTERFACE_DEF:
563            case TokenTypes.ENUM_DEF:
564            case TokenTypes.ANNOTATION_DEF:
565            case TokenTypes.RECORD_DEF:
566                final DetailAST classFrameNameIdent = ast.findFirstToken(TokenTypes.IDENT);
567                frameStack.addFirst(new ClassFrame(frame, classFrameNameIdent));
568                break;
569            case TokenTypes.SLIST:
570                frameStack.addFirst(new BlockFrame(frame, ast));
571                break;
572            case TokenTypes.METHOD_DEF:
573                final DetailAST methodFrameNameIdent = ast.findFirstToken(TokenTypes.IDENT);
574                final DetailAST mods = ast.findFirstToken(TokenTypes.MODIFIERS);
575                if (mods.findFirstToken(TokenTypes.LITERAL_STATIC) == null) {
576                    ((ClassFrame) frame).addInstanceMethod(methodFrameNameIdent);
577                }
578                else {
579                    ((ClassFrame) frame).addStaticMethod(methodFrameNameIdent);
580                }
581                frameStack.addFirst(new MethodFrame(frame, methodFrameNameIdent));
582                break;
583            case TokenTypes.CTOR_DEF:
584            case TokenTypes.COMPACT_CTOR_DEF:
585                final DetailAST ctorFrameNameIdent = ast.findFirstToken(TokenTypes.IDENT);
586                frameStack.addFirst(new ConstructorFrame(frame, ctorFrameNameIdent));
587                break;
588            case TokenTypes.ENUM_CONSTANT_DEF:
589                final DetailAST ident = ast.findFirstToken(TokenTypes.IDENT);
590                ((ClassFrame) frame).addStaticMember(ident);
591                break;
592            case TokenTypes.LITERAL_CATCH:
593                final AbstractFrame catchFrame = new CatchFrame(frame, ast);
594                catchFrame.addIdent(ast.findFirstToken(TokenTypes.PARAMETER_DEF).findFirstToken(
595                        TokenTypes.IDENT));
596                frameStack.addFirst(catchFrame);
597                break;
598            case TokenTypes.LITERAL_FOR:
599                final AbstractFrame forFrame = new ForFrame(frame, ast);
600                frameStack.addFirst(forFrame);
601                break;
602            case TokenTypes.LITERAL_NEW:
603                if (isAnonymousClassDef(ast)) {
604                    frameStack.addFirst(new AnonymousClassFrame(frame,
605                            ast.getFirstChild().toString()));
606                }
607                break;
608            default:
609                // do nothing
610        }
611    }

613    /**
614     * Collects variable declarations.
615     *
616     * @param ast variable token.
617     * @param frame current frame.
618     */
619    private static void collectVariableDeclarations(DetailAST ast, AbstractFrame frame) {
620        final DetailAST ident = ast.findFirstToken(TokenTypes.IDENT);
621        if (frame.getType() == FrameType.CLASS_FRAME) {
622            final DetailAST mods =
623                    ast.findFirstToken(TokenTypes.MODIFIERS);
624            if (ScopeUtil.isInInterfaceBlock(ast)
625                    || mods.findFirstToken(TokenTypes.LITERAL_STATIC) != null) {
626                ((ClassFrame) frame).addStaticMember(ident);
627            }
628            else {
629                ((ClassFrame) frame).addInstanceMember(ident);
630            }
631        }
632        else {
633            frame.addIdent(ident);
634        }
635    }

637    /**
638     * Ends parsing of the AST for declarations.
639     *
640     * @param frameStack Stack containing the FrameTree being built.
641     * @param ast AST that was parsed.
642     */
643    private void endCollectingDeclarations(Queue<AbstractFrame> frameStack, DetailAST ast) {
644        switch (ast.getType()) {
645            case TokenTypes.CLASS_DEF:
646            case TokenTypes.INTERFACE_DEF:
647            case TokenTypes.ENUM_DEF:
648            case TokenTypes.ANNOTATION_DEF:
649            case TokenTypes.SLIST:
650            case TokenTypes.METHOD_DEF:
651            case TokenTypes.CTOR_DEF:
652            case TokenTypes.LITERAL_CATCH:
653            case TokenTypes.LITERAL_FOR:
654            case TokenTypes.RECORD_DEF:
655            case TokenTypes.COMPACT_CTOR_DEF:
656                frames.put(ast, frameStack.poll());
657                break;
658            case TokenTypes.LITERAL_NEW:
659                if (isAnonymousClassDef(ast)) {
660                    frames.put(ast, frameStack.poll());
661                }
662                break;
663            default:
664                // do nothing
665        }
666    }

668    /**
669     * Whether the AST is a definition of an anonymous class.
670     *
671     * @param ast the AST to process.
672     * @return true if the AST is a definition of an anonymous class.
673     */
674    private static boolean isAnonymousClassDef(DetailAST ast) {
675        final DetailAST lastChild = ast.getLastChild();
676        return lastChild != null
677            && lastChild.getType() == TokenTypes.OBJBLOCK;
678    }

680    /**
681     * Returns the class frame where violation is found (where the field is used without 'this')
682     * or null otherwise.
683     *
684     * @param ast IDENT ast to check.
685     * @return the class frame where violation is found or null otherwise.
686     */
687    // -@cs[CyclomaticComplexity] Method already invokes too many methods that fully explain
688    // a logic, additional abstraction will not make logic/algorithm more readable.
689    private AbstractFrame getClassFrameWhereViolationIsFound(DetailAST ast) {
690        AbstractFrame frameWhereViolationIsFound = null;
691        final AbstractFrame variableDeclarationFrame = findFrame(ast, false);
692        final FrameType variableDeclarationFrameType = variableDeclarationFrame.getType();
693        final DetailAST prevSibling = ast.getPreviousSibling();
694        if (variableDeclarationFrameType == FrameType.CLASS_FRAME
695                && !validateOnlyOverlapping
696                && (prevSibling == null || !isInExpression(ast))
697                && canBeReferencedFromStaticContext(ast)) {
698            frameWhereViolationIsFound = variableDeclarationFrame;
699        }
700        else if (variableDeclarationFrameType == FrameType.METHOD_FRAME) {
701            if (isOverlappingByArgument(ast)) {
702                if (!isUserDefinedArrangementOfThis(variableDeclarationFrame, ast)
703                        && !isReturnedVariable(variableDeclarationFrame, ast)
704                        && canBeReferencedFromStaticContext(ast)
705                        && canAssignValueToClassField(ast)) {
706                    frameWhereViolationIsFound = findFrame(ast, true);
707                }
708            }
709            else if (!validateOnlyOverlapping
710                     && prevSibling == null
711                     && isAssignToken(ast.getParent().getType())
712                     && !isUserDefinedArrangementOfThis(variableDeclarationFrame, ast)
713                     && canBeReferencedFromStaticContext(ast)
714                     && canAssignValueToClassField(ast)) {
715                frameWhereViolationIsFound = findFrame(ast, true);
716            }
717        }
718        else if (variableDeclarationFrameType == FrameType.CTOR_FRAME
719                 && isOverlappingByArgument(ast)
720                 && !isUserDefinedArrangementOfThis(variableDeclarationFrame, ast)) {
721            frameWhereViolationIsFound = findFrame(ast, true);
722        }
723        else if (variableDeclarationFrameType == FrameType.BLOCK_FRAME
724                    && isOverlappingByLocalVariable(ast)
725                    && canAssignValueToClassField(ast)
726                    && !isUserDefinedArrangementOfThis(variableDeclarationFrame, ast)
727                    && !isReturnedVariable(variableDeclarationFrame, ast)
728                    && canBeReferencedFromStaticContext(ast)) {
729            frameWhereViolationIsFound = findFrame(ast, true);
730        }
731        return frameWhereViolationIsFound;
732    }

746     * Checks whether user arranges 'this' for variable in method, constructor, or block on his own.
747     *
748     * @param currentFrame current frame.
749     * @param ident ident token.
750     * @return true if user arranges 'this' for variable in method, constructor,
751     *         or block on his own.
752     */
753    private static boolean isUserDefinedArrangementOfThis(AbstractFrame currentFrame,
754                                                          DetailAST ident) {
755        final DetailAST blockFrameNameIdent = currentFrame.getFrameNameIdent();
756        final DetailAST definitionToken = blockFrameNameIdent.getParent();
757        final DetailAST blockStartToken = definitionToken.findFirstToken(TokenTypes.SLIST);
758        final DetailAST blockEndToken = getBlockEndToken(blockFrameNameIdent, blockStartToken);
759
760        boolean userDefinedArrangementOfThis = false;
761
762        final Set<DetailAST> variableUsagesInsideBlock =
763            getAllTokensWhichAreEqualToCurrent(definitionToken, ident,
764                blockEndToken.getLineNo());
765
766        for (DetailAST variableUsage : variableUsagesInsideBlock) {
767            final DetailAST prevSibling = variableUsage.getPreviousSibling();
768            if (prevSibling != null
769                    && prevSibling.getType() == TokenTypes.LITERAL_THIS) {
770                userDefinedArrangementOfThis = true;
771                break;
772            }
773        }
774        return userDefinedArrangementOfThis;
775    }

778     * Returns the token which ends the code block.
779     *
780     * @param blockNameIdent block name identifier.
781     * @param blockStartToken token which starts the block.
782     * @return the token which ends the code block.
783     */
784    private static DetailAST getBlockEndToken(DetailAST blockNameIdent, DetailAST blockStartToken) {
785        DetailAST blockEndToken = null;
786        final DetailAST blockNameIdentParent = blockNameIdent.getParent();
787        if (blockNameIdentParent.getType() == TokenTypes.CASE_GROUP) {
788            blockEndToken = blockNameIdentParent.getNextSibling();
789        }
790        else {
791            final Set<DetailAST> rcurlyTokens = getAllTokensOfType(blockNameIdent,
792                    TokenTypes.RCURLY);
793            for (DetailAST currentRcurly : rcurlyTokens) {
794                final DetailAST parent = currentRcurly.getParent();
795                if (TokenUtil.areOnSameLine(blockStartToken, parent)) {
796                    blockEndToken = currentRcurly;
797                }
798            }
799        }
800        return blockEndToken;
801    }

830     * Checks if the given {@code ast} is equal to the {@code tree} or a child of it.
831     *
832     * @param tree The tree to search.
833     * @param ast The AST to look for.
834     * @return {@code true} if the {@code ast} was found.
835     */
836    private static boolean isAstInside(DetailAST tree, DetailAST ast) {
837        boolean result = false;
838
839        if (isAstSimilar(tree, ast)) {
840            result = true;
841        }
842        else {
843            for (DetailAST child = tree.getFirstChild(); child != null
844                    && !result; child = child.getNextSibling()) {
845                result = isAstInside(child, ast);
846            }
847        }
848
849        return result;

852    /**
853     * Checks whether a field can be referenced from a static context.
854     *
855     * @param ident ident token.
856     * @return true if field can be referenced from a static context.
857     */
858    private boolean canBeReferencedFromStaticContext(DetailAST ident) {
859        AbstractFrame variableDeclarationFrame = findFrame(ident, false);
860        boolean staticInitializationBlock = false;
861        while (variableDeclarationFrame.getType() == FrameType.BLOCK_FRAME
862                || variableDeclarationFrame.getType() == FrameType.FOR_FRAME) {
863            final DetailAST blockFrameNameIdent = variableDeclarationFrame.getFrameNameIdent();
864            final DetailAST definitionToken = blockFrameNameIdent.getParent();
865            if (definitionToken.getType() == TokenTypes.STATIC_INIT) {
866                staticInitializationBlock = true;
867                break;
868            }
869            variableDeclarationFrame = variableDeclarationFrame.getParent();
870        }
871
872        boolean staticContext = false;
873        if (staticInitializationBlock) {
874            staticContext = true;
875        }
876        else {
877            if (variableDeclarationFrame.getType() == FrameType.CLASS_FRAME) {
878                final DetailAST codeBlockDefinition = getCodeBlockDefinitionToken(ident);
879                if (codeBlockDefinition != null) {
880                    final DetailAST modifiers = codeBlockDefinition.getFirstChild();
881                    staticContext = codeBlockDefinition.getType() == TokenTypes.STATIC_INIT
882                        || modifiers.findFirstToken(TokenTypes.LITERAL_STATIC) != null;
883                }
884            }
885            else {
886                final DetailAST frameNameIdent = variableDeclarationFrame.getFrameNameIdent();
887                final DetailAST definitionToken = frameNameIdent.getParent();
888                staticContext = definitionToken.findFirstToken(TokenTypes.MODIFIERS)
889                        .findFirstToken(TokenTypes.LITERAL_STATIC) != null;
890            }
891        }
892        return !staticContext;
893    }

993     * Collects all tokens of specific type starting with the current ast node.
994     *
995     * @param ast ast node.
996     * @param tokenType token type.
997     * @return a set of all tokens of specific type starting with the current ast node.
998     */
999    private static Set<DetailAST> getAllTokensOfType(DetailAST ast, int tokenType) {
1000        DetailAST vertex = ast;
1001        final Set<DetailAST> result = new HashSet<>();
1002        final Deque<DetailAST> stack = new ArrayDeque<>();
1003        while (vertex != null || !stack.isEmpty()) {
1004            if (!stack.isEmpty()) {
1005                vertex = stack.pop();
1006            }
1007            while (vertex != null) {
1008                if (vertex.getType() == tokenType) {
1009                    result.add(vertex);
1010                }
1011                if (vertex.getNextSibling() != null) {
1012                    stack.push(vertex.getNextSibling());
1013                }
1014                vertex = vertex.getFirstChild();
1015            }
1016        }
1017        return result;
1018    }

1022     * number is lower or equal to the end line number.
1023     *
1024     * @param ast ast node.

1026     * @param endLineNumber end line number.

1041                    && vertex.getLineNo() <= endLineNumber) {

1045                    stack.push(vertex.getNextSibling());

1055     * which line number is lower or equal to the end line number.
1056     *
1057     * @param ast ast node.
1058     * @param token token.
1059     * @param endLineNumber end line number.
1060     * @return a set of tokens which are equal to current token starting with the current ast node
1061     *         and which line number is lower or equal to the end line number.

1074                        && vertex.getLineNo() <= endLineNumber) {

1078                    stack.push(vertex.getNextSibling());
1079                }
1080                vertex = vertex.getFirstChild();
1081            }
1082        }
1083        return result;

1197    /**
1198     * Checks if the token is a Lambda parameter.
1199     *
1200     * @param ast the {@code DetailAST} value of the token to be checked
1201     * @return true if the token is a Lambda parameter
1202     */
1203    private static boolean isLambdaParameter(DetailAST ast) {
1204        DetailAST parent;
1205        for (parent = ast.getParent(); parent != null; parent = parent.getParent()) {
1206            if (parent.getType() == TokenTypes.LAMBDA) {
1207                break;
1208            }
1209        }
1210        final boolean isLambdaParameter;
1211        if (parent == null) {
1212            isLambdaParameter = false;
1213        }
1214        else if (ast.getType() == TokenTypes.PARAMETER_DEF) {
1215            isLambdaParameter = true;
1216        }
1217        else {
1218            final DetailAST lambdaParameters = parent.findFirstToken(TokenTypes.PARAMETERS);
1219            if (lambdaParameters == null) {
1220                isLambdaParameter = parent.getFirstChild().getText().equals(ast.getText());
1221            }
1222            else {
1223                isLambdaParameter = TokenUtil.findFirstTokenByPredicate(lambdaParameters,
1224                    paramDef -> {
1225                        final DetailAST param = paramDef.findFirstToken(TokenTypes.IDENT);
1226                        return param != null && param.getText().equals(ast.getText());
1227                    }).isPresent();
1228            }
1229        }
1230        return isLambdaParameter;
1231    }

1306         *
1307         * @return the parent frame
1308         */

1371            boolean result = false;
1372            for (DetailAST ast: set) {
1373                if (isProperDefinition(ident, ast)) {
1374                    result = true;
1375                    break;
1376                }
1377            }

1390            return nameToFind.equals(ast.getText())
1391                && CheckUtil.isBeforeInSource(ast, ident);
1392        }

1439    /**
1440     * A frame initiated at class, enum or interface definition; holds instance variable names.
1441     */
1442    private static class ClassFrame extends AbstractFrame {
1443
1444        /** Set of idents of instance members declared in this frame. */
1445        private final Set<DetailAST> instanceMembers;
1446        /** Set of idents of instance methods declared in this frame. */
1447        private final Set<DetailAST> instanceMethods;
1448        /** Set of idents of variables declared in this frame. */
1449        private final Set<DetailAST> staticMembers;
1450        /** Set of idents of static methods declared in this frame. */
1451        private final Set<DetailAST> staticMethods;
1452
1453        /**
1454         * Creates new instance of ClassFrame.
1455         *
1456         * @param parent parent frame.
1457         * @param ident frame name ident.
1458         */
1459        /* package */ ClassFrame(AbstractFrame parent, DetailAST ident) {
1460            super(parent, ident);
1461            instanceMembers = new HashSet<>();
1462            instanceMethods = new HashSet<>();
1463            staticMembers = new HashSet<>();
1464            staticMethods = new HashSet<>();
1465        }
1466
1467        @Override
1468        protected FrameType getType() {
1469            return FrameType.CLASS_FRAME;
1470        }
1471
1472        /**
1473         * Adds static member's ident.
1474         *
1475         * @param ident an ident of static member of the class.
1476         */
1477        public void addStaticMember(final DetailAST ident) {
1478            staticMembers.add(ident);
1479        }
1480
1481        /**
1482         * Adds static method's name.
1483         *
1484         * @param ident an ident of static method of the class.
1485         */
1486        public void addStaticMethod(final DetailAST ident) {
1487            staticMethods.add(ident);
1488        }
1489
1490        /**
1491         * Adds instance member's ident.
1492         *
1493         * @param ident an ident of instance member of the class.
1494         */
1495        public void addInstanceMember(final DetailAST ident) {
1496            instanceMembers.add(ident);
1497        }
1498
1499        /**
1500         * Adds instance method's name.
1501         *
1502         * @param ident an ident of instance method of the class.
1503         */
1504        public void addInstanceMethod(final DetailAST ident) {
1505            instanceMethods.add(ident);
1506        }
1507
1508        /**
1509         * Checks if a given name is a known instance member of the class.
1510         *
1511         * @param ident the IDENT ast of the name to check.
1512         * @return true is the given name is a name of a known
1513         *         instance member of the class.
1514         */
1515        public boolean hasInstanceMember(final DetailAST ident) {
1516            return containsFieldOrVariableDef(instanceMembers, ident);
1517        }
1518
1519        /**
1520         * Checks if a given name is a known instance method of the class.
1521         *
1522         * @param ident the IDENT ast of the method call to check.
1523         * @return true if the given ast is correspondent to a known
1524         *         instance method of the class.
1525         */
1526        public boolean hasInstanceMethod(final DetailAST ident) {
1527            return containsMethodDef(instanceMethods, ident);
1528        }
1529
1530        /**
1531         * Checks if a given name is a known static method of the class.
1532         *
1533         * @param ident the IDENT ast of the method call to check.
1534         * @return true is the given ast is correspondent to a known
1535         *         instance method of the class.
1536         */
1537        public boolean hasStaticMethod(final DetailAST ident) {
1538            return containsMethodDef(staticMethods, ident);
1539        }
1540
1541        /**
1542         * Checks whether given instance member has final modifier.
1543         *
1544         * @param instanceMember an instance member of a class.
1545         * @return true if given instance member has final modifier.
1546         */
1547        public boolean hasFinalField(final DetailAST instanceMember) {
1548            boolean result = false;
1549            for (DetailAST member : instanceMembers) {
1550                final DetailAST mods = member.getParent().findFirstToken(TokenTypes.MODIFIERS);
1551                final boolean finalMod = mods.findFirstToken(TokenTypes.FINAL) != null;
1552                if (finalMod && isAstSimilar(member, instanceMember)) {
1553                    result = true;
1554                    break;
1555                }
1556            }
1557            return result;
1558        }
1559
1560        @Override
1561        protected boolean containsFieldOrVariable(DetailAST nameToFind) {
1562            return containsFieldOrVariableDef(instanceMembers, nameToFind)
1563                    || containsFieldOrVariableDef(staticMembers, nameToFind);
1564        }
1565
1566        @Override
1567        protected boolean isProperDefinition(DetailAST ident, DetailAST ast) {
1568            final String nameToFind = ident.getText();
1569            return nameToFind.equals(ast.getText());
1570        }
1571
1572        @Override
1573        protected AbstractFrame getIfContains(DetailAST nameToFind, boolean lookForMethod) {
1574            AbstractFrame frame = null;
1575
1576            if (lookForMethod && containsMethod(nameToFind)
1577                || containsFieldOrVariable(nameToFind)) {
1578                frame = this;
1579            }
1580            else if (getParent() != null) {
1581                frame = getParent().getIfContains(nameToFind, lookForMethod);
1582            }
1583            return frame;
1584        }
1585
1586        /**
1587         * Check whether the frame contains a given method.
1588         *
1589         * @param methodToFind the AST of the method to find.
1590         * @return true, if a method with the same name and number of parameters is found.
1591         */
1592        private boolean containsMethod(DetailAST methodToFind) {
1593            return containsMethodDef(instanceMethods, methodToFind)
1594                || containsMethodDef(staticMethods, methodToFind);
1595        }
1596
1597        /**
1598         * Whether the set contains a method definition with the
1599         *     same name and number of parameters.
1600         *
1601         * @param set the set of definitions.
1602         * @param ident the specified method call IDENT ast.
1603         * @return true if the set contains a definition with the
1604         *     same name and number of parameters.
1605         */
1606        private static boolean containsMethodDef(Set<DetailAST> set, DetailAST ident) {
1607            boolean result = false;
1608            for (DetailAST ast: set) {
1609                if (isSimilarSignature(ident, ast)) {
1610                    result = true;
1611                    break;
1612                }
1613            }
1614            return result;
1615        }
1616
1617        /**
1618         * Whether the method definition has the same name and number of parameters.
1619         *
1620         * @param ident the specified method call IDENT ast.
1621         * @param ast the ast of a method definition to compare with.
1622         * @return true if a method definition has the same name and number of parameters
1623         *     as the method call.
1624         */
1625        private static boolean isSimilarSignature(DetailAST ident, DetailAST ast) {
1626            boolean result = false;
1627            final DetailAST elistToken = ident.getParent().findFirstToken(TokenTypes.ELIST);
1628            if (elistToken != null && ident.getText().equals(ast.getText())) {
1629                final int paramsNumber =
1630                    ast.getParent().findFirstToken(TokenTypes.PARAMETERS).getChildCount();
1631                final int argsNumber = elistToken.getChildCount();
1632                result = paramsNumber == argsNumber;
1633            }
1634            return result;
1635        }

1637    }
