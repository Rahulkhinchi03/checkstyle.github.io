035/**
036 * <p>
037 * Checks that
038 * <a href="https://www.oracle.com/technical-resources/articles/java/javadoc-tool.html#firstsentence">
039 * Javadoc summary sentence</a> does not contain phrases that are not recommended to use.
040 * Summaries that contain only the {@code {@inheritDoc}} tag are skipped.
041 * Check also violate Javadoc that does not contain first sentence.
042 * </p>
043 * <ul>
044 * <li>
045 * Property {@code violateExecutionOnNonTightHtml} - Control when to print violations
046 * if the Javadoc being examined by this check violates the tight html rules defined at
047 * <a href="https://checkstyle.org/writingjavadocchecks.html#Tight-HTML_rules">Tight-HTML Rules</a>.
048 * Type is {@code boolean}.
049 * Default value is {@code false}.
050 * </li>
051 * <li>
052 * Property {@code forbiddenSummaryFragments} - Specify the regexp for forbidden summary fragments.
053 * Type is {@code java.util.regex.Pattern}.
054 * Default value is {@code "^$"}.
055 * </li>
056 * <li>
057 * Property {@code period} - Specify the period symbol at the end of first javadoc sentence.
058 * Type is {@code java.lang.String}.
059 * Default value is {@code "."}.
060 * </li>
061 * </ul>
062 * <p>
063 * To configure the default check to validate that first sentence is not empty and first
064 * sentence is not missing:
065 * </p>
066 * <pre>
067 * &lt;module name=&quot;SummaryJavadocCheck&quot;/&gt;
068 * </pre>
069 * <p>
070 * Example of {@code {@inheritDoc}} without summary.
071 * </p>
072 * <pre>
073 * public class Test extends Exception {
074 * //Valid
075 *   &#47;**
076 *    * {&#64;inheritDoc}
077 *    *&#47;
078 *   public String ValidFunction(){
079 *     return "";
080 *   }
081 *   //Violation
082 *   &#47;**
083 *    *
084 *    *&#47;
085 *   public String InvalidFunction(){
086 *     return "";
087 *   }
088 * }
089 * </pre>
090 * <p>
091 * Example of non permitted empty javadoc for Inline Summary Javadoc.
092 * </p>
093 * <pre>
094 * public class Test extends Exception {
095 *   &#47;**
096 *    * {&#64;summary  }
097 *    *&#47;
098 *   public String InvalidFunctionOne(){ // violation
099 *     return "";
100 *   }
101 *
102 *   &#47;**
103 *    * {&#64;summary &lt;p&gt; &lt;p/&gt;}
104 *    *&#47;
105 *   public String InvalidFunctionTwo(){ // violation
106 *     return "";
107 *   }
108 *
109 *   &#47;**
110 *    * {&#64;summary &lt;p&gt;This is summary for validFunctionThree.&lt;p/&gt;}
111 *    *&#47;
112 *   public void validFunctionThree(){} // ok
113 * }
114 * </pre>
115 * <p>
116 * To ensure that summary do not contain phrase like "This method returns",
117 * use following config:
118 * </p>
119 * <pre>
120 * &lt;module name="SummaryJavadocCheck"&gt;
121 *   &lt;property name="forbiddenSummaryFragments"
122 *     value="^This method returns.*"/&gt;
123 * &lt;/module&gt;
124 * </pre>
125 * <p>
126 * To specify period symbol at the end of first javadoc sentence:
127 * </p>
128 * <pre>
129 * &lt;module name="SummaryJavadocCheck"&gt;
130 *   &lt;property name="period" value="。"/&gt;
131 * &lt;/module&gt;
132 * </pre>
133 * <p>
134 * Example of period property.
135 * </p>
136 * <pre>
137 * public class TestClass {
138 *  &#47;**
139 *   * This is invalid java doc.
140 *   *&#47;
141 *   void invalidJavaDocMethod() {
142 *   }
143 *  &#47;**
144 *   * This is valid java doc。
145 *   *&#47;
146 *   void validJavaDocMethod() {
147 *   }
148 * }
149 * </pre>
150 * <p>
151 * Example of period property for inline summary javadoc.
152 * </p>
153 * <pre>
154 * public class TestClass {
155 *  &#47;**
156 *   * {&#64;summary This is invalid java doc.}
157 *   *&#47;
158 *   public void invalidJavaDocMethod() { // violation
159 *   }
160 *  &#47;**
161 *   * {&#64;summary This is valid java doc。}
162 *   *&#47;
163 *   public void validJavaDocMethod() { // ok
164 *   }
165 * }
166 * </pre>
167 * <p>
168 * Example of inline summary javadoc with HTML tags.
169 * </p>
170 * <pre>
171 * public class Test {
172 *  &#47;**
173 *   * {&#64;summary First sentence is normally the summary.
174 *   * Use of html tags:
175 *   * &lt;ul&gt;
176 *   * &lt;li&gt;Item one.&lt;/li&gt;
177 *   * &lt;li&gt;Item two.&lt;/li&gt;
178 *   * &lt;/ul&gt;}
179 *   *&#47;
180 *   public void validInlineJavadoc() { // ok
181 *   }
182 * }
183 * </pre>
184 * <p>
185 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
186 * </p>
187 * <p>
188 * Violation Message Keys:
189 * </p>
190 * <ul>
191 * <li>
192 * {@code javadoc.missed.html.close}
193 * </li>
194 * <li>
195 * {@code javadoc.parse.rule.error}
196 * </li>
197 * <li>
198 * {@code javadoc.wrong.singleton.html.tag}
199 * </li>
200 * <li>
201 * {@code summary.first.sentence}
202 * </li>
203 * <li>
204 * {@code summary.javaDoc}
205 * </li>
206 * <li>
207 * {@code summary.javaDoc.missing}
208 * </li>
209 * <li>
210 * {@code summary.javaDoc.missing.period}
211 * </li>
212 * </ul>
213 *
214 * @since 6.0
215 */
216@StatelessCheck
217public class SummaryJavadocCheck extends AbstractJavadocCheck {
218
219    /**
220     * A key is pointing to the warning message text in "messages.properties"
221     * file.
222     */
223    public static final String MSG_SUMMARY_FIRST_SENTENCE = "summary.first.sentence";
224
225    /**
226     * A key is pointing to the warning message text in "messages.properties"
227     * file.
228     */
229    public static final String MSG_SUMMARY_JAVADOC = "summary.javaDoc";
230
231    /**
232     * A key is pointing to the warning message text in "messages.properties"
233     * file.
234     */
235    public static final String MSG_SUMMARY_JAVADOC_MISSING = "summary.javaDoc.missing";
236
237    /**
238     * A key is pointing to the warning message text in "messages.properties" file.
239     */
240    public static final String MSG_SUMMARY_MISSING_PERIOD = "summary.javaDoc.missing.period";
241
242    /**
243     * This regexp is used to convert multiline javadoc to single line without stars.
244     */
245    private static final Pattern JAVADOC_MULTILINE_TO_SINGLELINE_PATTERN =
246            Pattern.compile("\n[ ]+(\\*)|^[ ]+(\\*)");
247
248    /**
249     * This regexp is used to remove html tags, whitespace, and asterisks from a string.
250     */
251    private static final Pattern HTML_ELEMENTS =
252            Pattern.compile("<[^>]*>");
253
254    /**
255     * This regexp is used to extract the content of a summary javadoc tag.
256     */
257    private static final Pattern SUMMARY_PATTERN = Pattern.compile("\\{@summary ([\\S\\s]+)}");
258    /** Period literal. */
259    private static final String PERIOD = ".";
260
261    /** Summary tag text. */
262    private static final String SUMMARY_TEXT = "@summary";
263
264    /** Set of allowed Tokens tags in summary java doc. */
265    private static final Set<Integer> ALLOWED_TYPES = Set.of(
266                    JavadocTokenTypes.WS,
267                    JavadocTokenTypes.DESCRIPTION,
268                    JavadocTokenTypes.TEXT);
269
270    /**
271     * Specify the regexp for forbidden summary fragments.
272     */
273    private Pattern forbiddenSummaryFragments = CommonUtil.createPattern("^$");
274
275    /**
276     * Specify the period symbol at the end of first javadoc sentence.
277     */
278    private String period = PERIOD;
279
280    /**
281     * Setter to specify the regexp for forbidden summary fragments.
282     *
283     * @param pattern a pattern.
284     */
285    public void setForbiddenSummaryFragments(Pattern pattern) {
286        forbiddenSummaryFragments = pattern;
287    }
288
289    /**
290     * Setter to specify the period symbol at the end of first javadoc sentence.
291     *
292     * @param period period's value.
293     */
294    public void setPeriod(String period) {
295        this.period = period;
296    }

298    @Override
299    public int[] getDefaultJavadocTokens() {
300        return new int[] {
301            JavadocTokenTypes.JAVADOC,
302        };
303    }
304
305    @Override
306    public int[] getRequiredJavadocTokens() {
307        return getAcceptableJavadocTokens();
308    }

310    @Override
311    public void visitJavadocToken(DetailNode ast) {
312        if (containsSummaryTag(ast)) {
313            validateSummaryTag(ast);
314        }
315        else if (!startsWithInheritDoc(ast)) {
316            final String summaryDoc = getSummarySentence(ast);
317            if (summaryDoc.isEmpty()) {
318                log(ast.getLineNumber(), MSG_SUMMARY_JAVADOC_MISSING);
319            }
320            else if (!period.isEmpty()) {
321                final String firstSentence = getFirstSentence(ast);
322                final int endOfSentence = firstSentence.lastIndexOf(period);
323                if (!summaryDoc.contains(period)) {
324                    log(ast.getLineNumber(), MSG_SUMMARY_FIRST_SENTENCE);
325                }
326                if (endOfSentence != -1
327                        && containsForbiddenFragment(firstSentence.substring(0, endOfSentence))) {
328                    log(ast.getLineNumber(), MSG_SUMMARY_JAVADOC);
329                }
330            }
331        }
332    }
333
334    /**
335     * Checks if summary tag present.
336     *
337     * @param javadoc javadoc root node.
338     * @return {@code true} if first sentence contains @summary tag.
339     */
340    private static boolean containsSummaryTag(DetailNode javadoc) {
341        final Optional<DetailNode> node = Arrays.stream(javadoc.getChildren())
342                .filter(SummaryJavadocCheck::isInlineTagPresent)
343                .findFirst()
344                .map(SummaryJavadocCheck::getInlineTagNodeWithinHtmlElement);
345
346        return node.isPresent() && isSummaryTag(node.get());
347    }
348
349    /**
350     * Checks if the inline tag node is present.
351     *
352     * @param ast ast node to check.
353     * @return true, if the inline tag node is present.
354     */
355    private static boolean isInlineTagPresent(DetailNode ast) {
356        return ast.getType() == JavadocTokenTypes.JAVADOC_INLINE_TAG
357                || ast.getType() == JavadocTokenTypes.HTML_ELEMENT
358                && getInlineTagNodeWithinHtmlElement(ast) != null;
359    }
360
361    /**
362     * Returns an inline javadoc tag node that is within a html tag.
363     *
364     * @param ast html tag node.
365     * @return inline summary javadoc tag node or null if no node is found.
366     */
367    private static DetailNode getInlineTagNodeWithinHtmlElement(DetailNode ast) {
368        DetailNode node = ast;
369        DetailNode result = null;
370        // node can never be null as this method is called when there is a HTML_ELEMENT
371        if (node.getType() == JavadocTokenTypes.JAVADOC_INLINE_TAG) {
372            result = node;
373        }
374        else if (node.getType() == JavadocTokenTypes.HTML_TAG) {
375            // HTML_TAG always has more than 2 children.
376            node = node.getChildren()[1];
377            result = getInlineTagNodeWithinHtmlElement(node);
378        }
379        else if (node.getType() == JavadocTokenTypes.HTML_ELEMENT
380                // Condition for SINGLETON html element which cannot contain summary node
381                && node.getChildren()[0].getChildren().length > 1) {
382            // Html elements have one tested tag before actual content inside it
383            node = node.getChildren()[0].getChildren()[1];
384            result = getInlineTagNodeWithinHtmlElement(node);
385        }
386        return result;
387    }
388
389    /**
390     * Checks if the first tag inside ast is summary tag.
391     *
392     * @param javadoc root node.
393     * @return {@code true} if first tag is summary tag.
394     */
395    private static boolean isSummaryTag(DetailNode javadoc) {
396        final DetailNode[] child = javadoc.getChildren();
397
398        // Checking size of ast is not required, since ast contains
399        // children of Inline Tag, as at least 2 children will be present which are
400        // RCURLY and LCURLY.
401        return child[1].getType() == JavadocTokenTypes.CUSTOM_NAME
402                && SUMMARY_TEXT.equals(child[1].getText());
403    }
404
405    /**
406     * Checks the inline summary (if present) for {@code period} at end and forbidden fragments.
407     *
408     * @param ast javadoc root node.
409     */
410    private void validateSummaryTag(DetailNode ast) {
411        final String inlineSummary = getInlineSummary();
412        final String summaryVisible = getVisibleContent(inlineSummary);
413        if (summaryVisible.isEmpty()) {
414            log(ast.getLineNumber(), MSG_SUMMARY_JAVADOC_MISSING);
415        }
416        else if (!period.isEmpty()) {
417            if (isPeriodAtEnd(summaryVisible, period)) {
418                log(ast.getLineNumber(), MSG_SUMMARY_MISSING_PERIOD);
419            }
420            else if (containsForbiddenFragment(inlineSummary)) {
421                log(ast.getLineNumber(), MSG_SUMMARY_JAVADOC);
422            }
423        }
424    }
425
426    /**
427     * Gets entire content of summary tag.
428     *
429     * @return summary sentence of javadoc root node.
430     */
431    private String getInlineSummary() {
432        final DetailAST blockCommentAst = getBlockCommentAst();
433        final String javadocText = blockCommentAst.getFirstChild().getText();
434        final Matcher matcher = SUMMARY_PATTERN.matcher(javadocText);
435        String comment = "";
436        if (matcher.find()) {
437            comment = matcher.group(1);
438        }
439        return JAVADOC_MULTILINE_TO_SINGLELINE_PATTERN.matcher(comment)
440                .replaceAll("");
441    }
442
443    /**
444     * Gets the string that is visible to user in javadoc.
445     *
446     * @param summary entire content of summary javadoc.
447     * @return string that is visible to user in javadoc.
448     */
449    private static String getVisibleContent(String summary) {
450        final String visibleSummary = HTML_ELEMENTS.matcher(summary).replaceAll("");
451        return visibleSummary.trim();
452    }
453
454    /**
455     * Checks if the string ends with period.
456     *
457     * @param sentence string to check for period at end.
458     * @param period string to check within sentence.
459     * @return {@code true} if sentence ends with period.
460     */
461    private static boolean isPeriodAtEnd(String sentence, String period) {
462        final String summarySentence = sentence.trim();
463        return summarySentence.lastIndexOf(period) != summarySentence.length() - 1;
464    }
465
466    /**
467     * Tests if first sentence contains forbidden summary fragment.
468     *
469     * @param firstSentence string with first sentence.
470     * @return {@code true} if first sentence contains forbidden summary fragment.
471     */
472    private boolean containsForbiddenFragment(String firstSentence) {
473        final String javadocText = JAVADOC_MULTILINE_TO_SINGLELINE_PATTERN
474                .matcher(firstSentence).replaceAll(" ").trim();
475        return forbiddenSummaryFragments.matcher(trimExcessWhitespaces(javadocText)).find();
476    }
477
478    /**
479     * Trims the given {@code text} of duplicate whitespaces.
480     *
481     * @param text the text to transform.
482     * @return the finalized form of the text.
483     */
484    private static String trimExcessWhitespaces(String text) {
485        final StringBuilder result = new StringBuilder(256);
486        boolean previousWhitespace = true;
487
488        for (char letter : text.toCharArray()) {
489            final char print;
490            if (Character.isWhitespace(letter)) {
491                if (previousWhitespace) {
492                    continue;
493                }
494
495                previousWhitespace = true;
496                print = ' ';
497            }
498            else {
499                previousWhitespace = false;
500                print = letter;
501            }
502
503            result.append(print);
504        }
505
506        return result.toString();
507    }

509    /**
510     * Checks if the node starts with an {&#64;inheritDoc}.
511     *
512     * @param root the root node to examine.
513     * @return {@code true} if the javadoc starts with an {&#64;inheritDoc}.
514     */
515    private static boolean startsWithInheritDoc(DetailNode root) {
516        boolean found = false;
517        final DetailNode[] children = root.getChildren();
518
519        for (int i = 0; !found; i++) {
520            final DetailNode child = children[i];
521            if (child.getType() == JavadocTokenTypes.JAVADOC_INLINE_TAG
522                    && child.getChildren()[1].getType() == JavadocTokenTypes.INHERIT_DOC_LITERAL) {
523                found = true;
524            }
525            else if (child.getType() != JavadocTokenTypes.LEADING_ASTERISK
526                    && !CommonUtil.isBlank(child.getText())) {
527                break;
528            }
529        }
530
531        return found;
532    }
533
534    /**
535     * Finds and returns summary sentence.
536     *
537     * @param ast javadoc root node.
538     * @return violation string.
539     */
540    private static String getSummarySentence(DetailNode ast) {
541        boolean flag = true;
542        final StringBuilder result = new StringBuilder(256);
543        for (DetailNode child : ast.getChildren()) {
544            if (ALLOWED_TYPES.contains(child.getType())) {
545                result.append(child.getText());
546            }
547            else if (child.getType() == JavadocTokenTypes.HTML_ELEMENT
548                    && CommonUtil.isBlank(result.toString().trim())) {
549                result.append(getStringInsideTag(result.toString(),
550                        child.getChildren()[0].getChildren()[0]));
551            }
552            else if (child.getType() == JavadocTokenTypes.JAVADOC_TAG) {
553                flag = false;
554            }
555            if (!flag) {
556                break;
557            }
558        }
559        return result.toString().trim();
560    }
561
562    /**
563     * Get concatenated string within text of html tags.
564     *
565     * @param result javadoc string
566     * @param detailNode javadoc tag node
567     * @return java doc tag content appended in result
568     */
569    private static String getStringInsideTag(String result, DetailNode detailNode) {
570        final StringBuilder contents = new StringBuilder(result);
571        DetailNode tempNode = detailNode;
572        while (tempNode != null) {
573            if (tempNode.getType() == JavadocTokenTypes.TEXT) {
574                contents.append(tempNode.getText());
575            }
576            tempNode = JavadocUtil.getNextSibling(tempNode);
577        }
578        return contents.toString();
579    }
580
581    /**
582     * Finds and returns first sentence.
583     *
584     * @param ast Javadoc root node.
585     * @return first sentence.
586     */
587    private static String getFirstSentence(DetailNode ast) {
588        final StringBuilder result = new StringBuilder(256);
589        final String periodSuffix = PERIOD + ' ';
590        for (DetailNode child : ast.getChildren()) {
591            final String text;
592            if (child.getChildren().length == 0) {
593                text = child.getText();
594            }
595            else {
596                text = getFirstSentence(child);
597            }
598
599            if (text.contains(periodSuffix)) {
600                result.append(text, 0, text.indexOf(periodSuffix) + 1);
601                break;
602            }
603
604            result.append(text);
605        }
606        return result.toString();
607    }
