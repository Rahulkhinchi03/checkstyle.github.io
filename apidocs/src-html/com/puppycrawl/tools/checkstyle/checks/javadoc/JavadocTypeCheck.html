042/**
043 * <p>
044 * Checks the Javadoc comments for type definitions. By default, does
045 * not check for author or version tags. The scope to verify is specified using the {@code Scope}
046 * class and defaults to {@code Scope.PRIVATE}. To verify another scope, set property
047 * scope to one of the {@code Scope} constants. To define the format for an author
048 * tag or a version tag, set property authorFormat or versionFormat respectively to a
049 * <a href="https://docs.oracle.com/javase/7/docs/api/java/util/regex/Pattern.html">
050 * pattern</a>.
051 * </p>
052 * <p>
053 * Does not perform checks for author and version tags for inner classes,
054 * as they should be redundant because of outer class.
055 * </p>
056 * <p>
057 * Error messages about type parameters and record components for which no param tags are present
058 * can be suppressed by defining property {@code allowMissingParamTags}.
059 * </p>
060 * <ul>
061 * <li>
062 * Property {@code scope} - Specify the visibility scope where Javadoc comments are checked.
063 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
064 * Default value is {@code private}.
065 * </li>
066 * <li>
067 * Property {@code excludeScope} - Specify the visibility scope where Javadoc
068 * comments are not checked.
069 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
070 * Default value is {@code null}.
071 * </li>
072 * <li>
073 * Property {@code authorFormat} - Specify the pattern for {@code @author} tag.
074 * Type is {@code java.util.regex.Pattern}.
075 * Default value is {@code null}.
076 * </li>
077 * <li>
078 * Property {@code versionFormat} - Specify the pattern for {@code @version} tag.
079 * Type is {@code java.util.regex.Pattern}.
080 * Default value is {@code null}.
081 * </li>
082 * <li>
083 * Property {@code allowMissingParamTags} - Control whether to ignore violations
084 * when a class has type parameters but does not have matching param tags in the Javadoc.
085 * Type is {@code boolean}.
086 * Default value is {@code false}.
087 * </li>
088 * <li>
089 * Property {@code allowUnknownTags} - Control whether to ignore violations when
090 * a Javadoc tag is not recognised.
091 * Type is {@code boolean}.
092 * Default value is {@code false}.
093 * </li>
094 * <li>
095 * Property {@code allowedAnnotations} - Specify annotations that allow
096 * missed documentation. Only short names are allowed, e.g. {@code Generated}.
097 * Type is {@code java.lang.String[]}.
098 * Default value is {@code Generated}.
099 * </li>
100 * <li>
101 * Property {@code tokens} - tokens to check
102 * Type is {@code java.lang.String[]}.
103 * Validation type is {@code tokenSet}.
104 * Default value is:
105 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
106 * INTERFACE_DEF</a>,
107 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
108 * CLASS_DEF</a>,
109 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
110 * ENUM_DEF</a>,
111 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_DEF">
112 * ANNOTATION_DEF</a>,
113 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#RECORD_DEF">
114 * RECORD_DEF</a>.
115 * </li>
116 * </ul>
117 * <p>
118 * To configure the default check:
119 * </p>
120 * <pre>
121 * &lt;module name="JavadocType"/&gt;
122 * </pre>
123 * <p>
124 * To configure the check for {@code public} scope:
125 * </p>
126 * <pre>
127 * &lt;module name="JavadocType"&gt;
128 *   &lt;property name="scope" value="public"/&gt;
129 * &lt;/module&gt;
130 * </pre>
131 * <p>
132 * To configure the check for an {@code @author} tag:
133 * </p>
134 * <pre>
135 * &lt;module name="JavadocType"&gt;
136 *   &lt;property name="authorFormat" value="\S"/&gt;
137 * &lt;/module&gt;
138 * </pre>
139 * <p>
140 * To configure the check for a CVS revision version tag:
141 * </p>
142 * <pre>
143 * &lt;module name="JavadocType"&gt;
144 *   &lt;property name="versionFormat" value="\$Revision.*\$"/&gt;
145 * &lt;/module&gt;
146 * </pre>
147 * <p>
148 * To configure the check for {@code private} classes only:
149 * </p>
150 * <pre>
151 * &lt;module name="JavadocType"&gt;
152 *   &lt;property name="scope" value="private"/&gt;
153 *   &lt;property name="excludeScope" value="package"/&gt;
154 * &lt;/module&gt;
155 * </pre>
156 * <p>
157 * Example that allows missing comments for classes annotated with
158 * {@code @SpringBootApplication} and {@code @Configuration}:
159 * </p>
160 * <pre>
161 * &#64;SpringBootApplication // no violations about missing comment on class
162 * public class Application {}
163 *
164 * &#64;Configuration // no violations about missing comment on class
165 * class DatabaseConfiguration {}
166 * </pre>
167 * <p>
168 * Use following configuration:
169 * </p>
170 * <pre>
171 * &lt;module name="JavadocType"&gt;
172 *   &lt;property name="allowedAnnotations" value="SpringBootApplication,Configuration"/&gt;
173 * &lt;/module&gt;
174 * </pre>
175 * <p>
176 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
177 * </p>
178 * <p>
179 * Violation Message Keys:
180 * </p>
181 * <ul>
182 * <li>
183 * {@code javadoc.unknownTag}
184 * </li>
185 * <li>
186 * {@code javadoc.unusedTag}
187 * </li>
188 * <li>
189 * {@code javadoc.unusedTagGeneral}
190 * </li>
191 * <li>
192 * {@code type.missingTag}
193 * </li>
194 * <li>
195 * {@code type.tagFormat}
196 * </li>
197 * </ul>
198 *
199 * @since 3.0
200 *
201 */
202@StatelessCheck

236    /** Open angle bracket literal. */
237    private static final String OPEN_ANGLE_BRACKET = "<";
238
239    /** Close angle bracket literal. */

245    /** Pattern to match type name within angle brackets in javadoc param tag. */
246    private static final Pattern TYPE_NAME_IN_JAVADOC_TAG =
247            Pattern.compile("\\s*<([^>]+)>.*");
248
249    /** Pattern to split type name field in javadoc param tag. */
250    private static final Pattern TYPE_NAME_IN_JAVADOC_TAG_SPLITTER =
251            Pattern.compile("\\s+");

361    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
362    @SuppressWarnings("deprecation")
363    @Override
364    public void visitToken(DetailAST ast) {
365        if (shouldCheck(ast)) {
366            final FileContents contents = getFileContents();
367            final int lineNo = ast.getLineNo();
368            final TextBlock textBlock = contents.getJavadocBefore(lineNo);
369            if (textBlock != null) {
370                final List<JavadocTag> tags = getJavadocTags(textBlock);
371                if (ScopeUtil.isOuterMostType(ast)) {
372                    // don't check author/version for inner classes
373                    checkTag(ast, tags, JavadocTagInfo.AUTHOR.getName(),
374                            authorFormat);
375                    checkTag(ast, tags, JavadocTagInfo.VERSION.getName(),
376                            versionFormat);
377                }
378
379                final List<String> typeParamNames =
380                    CheckUtil.getTypeParameterNames(ast);
381                final List<String> recordComponentNames =
382                    getRecordComponentNames(ast);
383
384                if (!allowMissingParamTags) {
385
386                    typeParamNames.forEach(typeParamName -> {
387                        checkTypeParamTag(ast, tags, typeParamName);
388                    });
389
390                    recordComponentNames.forEach(componentName -> {
391                        checkComponentParamTag(ast, tags, componentName);
392                    });
393                }
394
395                checkUnusedParamTags(tags, typeParamNames, recordComponentNames);
396            }
397        }
398    }

400    /**
401     * Whether we should check this node.
402     *
403     * @param ast a given node.
404     * @return whether we should check a given node.
405     */
406    private boolean shouldCheck(DetailAST ast) {
407        final Scope customScope = ScopeUtil.getScope(ast);
408        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
409
410        return customScope.isIn(scope)
411            && (surroundingScope == null || surroundingScope.isIn(scope))
412            && (excludeScope == null
413                || !customScope.isIn(excludeScope)
414                || surroundingScope != null
415                && !surroundingScope.isIn(excludeScope))
416            && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
417    }

419    /**
420     * Gets all standalone tags from a given javadoc.
421     *
422     * @param textBlock the Javadoc comment to process.
423     * @return all standalone tags from the given javadoc.
424     */
425    private List<JavadocTag> getJavadocTags(TextBlock textBlock) {
426        final JavadocTags tags = JavadocUtil.getJavadocTags(textBlock,
427            JavadocUtil.JavadocTagType.BLOCK);
428        if (!allowUnknownTags) {
429            for (final InvalidJavadocTag tag : tags.getInvalidTags()) {
430                log(tag.getLine(), tag.getCol(), MSG_UNKNOWN_TAG,
431                    tag.getName());
432            }
433        }
434        return tags.getValidTags();
435    }

437    /**
438     * Verifies that a type definition has a required tag.
439     *
440     * @param ast the AST node for the type definition.
441     * @param tags tags from the Javadoc comment for the type definition.
442     * @param tagName the required tag name.
443     * @param formatPattern regexp for the tag value.
444     */
445    private void checkTag(DetailAST ast, List<JavadocTag> tags, String tagName,
446                          Pattern formatPattern) {
447        if (formatPattern != null) {
448            boolean hasTag = false;
449            final String tagPrefix = "@";
450
451            for (final JavadocTag tag :tags) {
452                if (tag.getTagName().equals(tagName)) {
453                    hasTag = true;
454                    if (!formatPattern.matcher(tag.getFirstArg()).find()) {
455                        log(ast, MSG_TAG_FORMAT, tagPrefix + tagName, formatPattern.pattern());
456                    }
457                }
458            }
459            if (!hasTag) {
460                log(ast, MSG_MISSING_TAG, tagPrefix + tagName);
461            }
462        }
463    }

465    /**
466     * Verifies that a record definition has the specified param tag for
467     * the specified record component name.
468     *
469     * @param ast the AST node for the record definition.
470     * @param tags tags from the Javadoc comment for the record definition.
471     * @param recordComponentName the name of the type parameter
472     */
473    private void checkComponentParamTag(DetailAST ast,
474                                        List<JavadocTag> tags,
475                                        String recordComponentName) {
476
477        final boolean found = tags
478            .stream()
479            .filter(JavadocTag::isParamTag)
480            .anyMatch(tag -> tag.getFirstArg().indexOf(recordComponentName) == 0);
481
482        if (!found) {
483            log(ast, MSG_MISSING_TAG, JavadocTagInfo.PARAM.getText()
484                + SPACE + recordComponentName);
485        }
486    }
487
488    /**
489     * Verifies that a type definition has the specified param tag for
490     * the specified type parameter name.
491     *
492     * @param ast the AST node for the type definition.
493     * @param tags tags from the Javadoc comment for the type definition.
494     * @param typeParamName the name of the type parameter
495     */
496    private void checkTypeParamTag(DetailAST ast,
497            List<JavadocTag> tags, String typeParamName) {
498        final String typeParamNameWithBrackets =
499            OPEN_ANGLE_BRACKET + typeParamName + CLOSE_ANGLE_BRACKET;
500
501        final boolean found = tags
502            .stream()
503            .filter(JavadocTag::isParamTag)
504            .anyMatch(tag -> tag.getFirstArg().indexOf(typeParamNameWithBrackets) == 0);
505
506        if (!found) {
507            log(ast, MSG_MISSING_TAG, JavadocTagInfo.PARAM.getText()
508                + SPACE + typeParamNameWithBrackets);
509        }
510    }
511
512    /**
513     * Checks for unused param tags for type parameters and record components.
514     *
515     * @param tags tags from the Javadoc comment for the type definition
516     * @param typeParamNames names of type parameters
517     * @param recordComponentNames record component names in this definition
518     */
519    private void checkUnusedParamTags(
520        List<JavadocTag> tags,
521        List<String> typeParamNames,
522        List<String> recordComponentNames) {
523
524        for (final JavadocTag tag: tags) {
525            if (tag.isParamTag()) {
526                final String paramName = extractParamNameFromTag(tag);
527                final boolean found = typeParamNames.contains(paramName)
528                        || recordComponentNames.contains(paramName);
529
530                if (!found) {
531                    final String actualParamName =
532                        TYPE_NAME_IN_JAVADOC_TAG_SPLITTER.split(tag.getFirstArg())[0];
533                    log(tag.getLineNo(), tag.getColumnNo(),
534                        MSG_UNUSED_TAG,
535                        JavadocTagInfo.PARAM.getText(), actualParamName);
536                }
537            }
538        }
539
540    }

542    /**
543     * Extracts parameter name from tag.
544     *
545     * @param tag javadoc tag to extract parameter name
546     * @return extracts type parameter name from tag
547     */
548    private static String extractParamNameFromTag(JavadocTag tag) {
549        final String typeParamName;
550        final Matcher matchInAngleBrackets =
551                TYPE_NAME_IN_JAVADOC_TAG.matcher(tag.getFirstArg());
552        if (matchInAngleBrackets.find()) {
553            typeParamName = matchInAngleBrackets.group(1).trim();
554        }
555        else {
556            typeParamName = TYPE_NAME_IN_JAVADOC_TAG_SPLITTER.split(tag.getFirstArg())[0];
557        }
558        return typeParamName;
559    }

561    /**
562     * Collects the record components in a record definition.
563     *
564     * @param node the possible record definition ast.
565     * @return the record components in this record definition.
566     */
567    private static List<String> getRecordComponentNames(DetailAST node) {
568        final DetailAST components = node.findFirstToken(TokenTypes.RECORD_COMPONENTS);
569        final List<String> componentList = new ArrayList<>();
570
571        if (components != null) {
572            TokenUtil.forEachChild(components,
573                TokenTypes.RECORD_COMPONENT_DEF, component -> {
574                    final DetailAST ident = component.findFirstToken(TokenTypes.IDENT);
575                    componentList.add(ident.getText());
576                });
577        }
578
579        return componentList;
580    }
