285    @Override
286    public void execute() {
287        final long startTime = System.currentTimeMillis();
288
289        try {
290            final String version = CheckstyleAntTask.class.getPackage().getImplementationVersion();
291
292            log("checkstyle version " + version, Project.MSG_VERBOSE);
293
294            // Check for no arguments
295            if (fileName == null
296                    && fileSets.isEmpty()
297                    && paths.isEmpty()) {
298                throw new BuildException(
299                        "Must specify at least one of 'file' or nested 'fileset' or 'path'.",
300                        getLocation());
301            }
302            if (config == null) {
303                throw new BuildException("Must specify 'config'.", getLocation());
304            }
305            realExecute(version);
306        }
307        finally {
308            final long endTime = System.currentTimeMillis();
309            log("Total execution took " + (endTime - startTime) + TIME_SUFFIX,
310                Project.MSG_VERBOSE);
311        }
312    }
313
314    /**
315     * Helper implementation to perform execution.
316     *
317     * @param checkstyleVersion Checkstyle compile version.
318     */
319    private void realExecute(String checkstyleVersion) {
320        // Create the root module
321        RootModule rootModule = null;
322        try {
323            rootModule = createRootModule();
324
325            // setup the listeners
326            final AuditListener[] listeners = getListeners();
327            for (AuditListener element : listeners) {
328                rootModule.addListener(element);
329            }
330            final SeverityLevelCounter warningCounter =
331                new SeverityLevelCounter(SeverityLevel.WARNING);
332            rootModule.addListener(warningCounter);
333
334            processFiles(rootModule, warningCounter, checkstyleVersion);
335        }
336        finally {
337            if (rootModule != null) {
338                rootModule.destroy();
339            }
340        }
341    }

343    /**
344     * Scans and processes files by means given root module.
345     *
346     * @param rootModule Root module to process files
347     * @param warningCounter Root Module's counter of warnings
348     * @param checkstyleVersion Checkstyle compile version
349     * @throws BuildException if the files could not be processed,
350     *     or if the build failed due to violations.
351     */
352    private void processFiles(RootModule rootModule, final SeverityLevelCounter warningCounter,
353            final String checkstyleVersion) {
354        final long startTime = System.currentTimeMillis();
355        final List<File> files = getFilesToCheck();
356        final long endTime = System.currentTimeMillis();
357        log("To locate the files took " + (endTime - startTime) + TIME_SUFFIX,
358            Project.MSG_VERBOSE);
359
360        log("Running Checkstyle "
361                + Objects.toString(checkstyleVersion, "")
362                + " on " + files.size()
363                + " files", Project.MSG_INFO);
364        log("Using configuration " + config, Project.MSG_VERBOSE);
365
366        final int numErrs;
367
368        try {
369            final long processingStartTime = System.currentTimeMillis();
370            numErrs = rootModule.process(files);
371            final long processingEndTime = System.currentTimeMillis();
372            log("To process the files took " + (processingEndTime - processingStartTime)
373                + TIME_SUFFIX, Project.MSG_VERBOSE);
374        }
375        catch (CheckstyleException ex) {
376            throw new BuildException("Unable to process files: " + files, ex);
377        }
378        final int numWarnings = warningCounter.getCount();
379        final boolean okStatus = numErrs <= maxErrors && numWarnings <= maxWarnings;
380
381        // Handle the return status
382        if (!okStatus) {
383            final String failureMsg =
384                    "Got " + numErrs + " errors and " + numWarnings
385                            + " warnings.";
386            if (failureProperty != null) {
387                getProject().setProperty(failureProperty, failureMsg);
388            }
389
390            if (failOnViolation) {
391                throw new BuildException(failureMsg, getLocation());
392            }
393        }
394    }

396    /**
397     * Creates new instance of the root module.
398     *
399     * @return new instance of the root module
400     * @throws BuildException if the root module could not be created.
401     */
402    private RootModule createRootModule() {
403        final RootModule rootModule;
404        try {
405            final Properties props = createOverridingProperties();
406            final ThreadModeSettings threadModeSettings =
407                    ThreadModeSettings.SINGLE_THREAD_MODE_INSTANCE;
408            final ConfigurationLoader.IgnoredModulesOptions ignoredModulesOptions;
409            if (executeIgnoredModules) {
410                ignoredModulesOptions = ConfigurationLoader.IgnoredModulesOptions.EXECUTE;
411            }
412            else {
413                ignoredModulesOptions = ConfigurationLoader.IgnoredModulesOptions.OMIT;
414            }
415
416            final Configuration configuration = ConfigurationLoader.loadConfiguration(config,
417                    new PropertiesExpander(props), ignoredModulesOptions, threadModeSettings);
418
419            final ClassLoader moduleClassLoader =
420                Checker.class.getClassLoader();
421
422            final ModuleFactory factory = new PackageObjectFactory(
423                    Checker.class.getPackage().getName() + ".", moduleClassLoader);
424
425            rootModule = (RootModule) factory.createModule(configuration.getName());
426            rootModule.setModuleClassLoader(moduleClassLoader);
427            rootModule.configure(configuration);
428        }
429        catch (final CheckstyleException ex) {
430            throw new BuildException(String.format(Locale.ROOT, "Unable to create Root Module: "
431                    + "config {%s}, classpath {%s}.", config, classpath), ex);
432        }
433        return rootModule;
434    }

436    /**
437     * Create the Properties object based on the arguments specified
438     * to the ANT task.
439     *
440     * @return the properties for property expansion
441     * @throws BuildException if the properties file could not be loaded.
442     */
443    private Properties createOverridingProperties() {
444        final Properties returnValue = new Properties();
445
446        // Load the properties file if specified
447        if (properties != null) {
448            try (InputStream inStream = Files.newInputStream(properties.toPath())) {
449                returnValue.load(inStream);
450            }
451            catch (final IOException ex) {
452                throw new BuildException("Error loading Properties file '"
453                        + properties + "'", ex, getLocation());
454            }
455        }
456
457        // override with Ant properties like ${basedir}
458        final Map<String, Object> antProps = getProject().getProperties();
459        for (Map.Entry<String, Object> entry : antProps.entrySet()) {
460            final String value = String.valueOf(entry.getValue());
461            returnValue.setProperty(entry.getKey(), value);
462        }
463
464        // override with properties specified in subelements
465        for (Property p : overrideProps) {
466            returnValue.setProperty(p.getKey(), p.getValue());
467        }
468
469        return returnValue;
470    }

472    /**
473     * Return the list of listeners set in this task.
474     *
475     * @return the list of listeners.
476     * @throws BuildException if the listeners could not be created.
477     */
478    private AuditListener[] getListeners() {
479        final int formatterCount = Math.max(1, formatters.size());
480
481        final AuditListener[] listeners = new AuditListener[formatterCount];
482
483        // formatters
484        try {
485            if (formatters.isEmpty()) {
486                final OutputStream debug = new LogOutputStream(this, Project.MSG_DEBUG);
487                final OutputStream err = new LogOutputStream(this, Project.MSG_ERR);
488                listeners[0] = new DefaultLogger(debug, AutomaticBean.OutputStreamOptions.CLOSE,
489                        err, AutomaticBean.OutputStreamOptions.CLOSE);
490            }
491            else {
492                for (int i = 0; i < formatterCount; i++) {
493                    final Formatter formatter = formatters.get(i);
494                    listeners[i] = formatter.createListener(this);
495                }
496            }
497        }
498        catch (IOException ex) {
499            throw new BuildException(String.format(Locale.ROOT, "Unable to create listeners: "
500                    + "formatters {%s}.", formatters), ex);
501        }
502        return listeners;
503    }

505    /**
506     * Returns the list of files (full path name) to process.
507     *
508     * @return the list of files included via the fileName, filesets and paths.
509     */
510    private List<File> getFilesToCheck() {
511        final List<File> allFiles = new ArrayList<>();
512        if (fileName != null) {
513            // oops, we've got an additional one to process, don't
514            // forget it. No sweat, it's fully resolved via the setter.
515            log("Adding standalone file for audit", Project.MSG_VERBOSE);
516            allFiles.add(new File(fileName));
517        }
518
519        final List<File> filesFromFileSets = scanFileSets();
520        allFiles.addAll(filesFromFileSets);
521
522        final List<File> filesFromPaths = scanPaths();
523        allFiles.addAll(filesFromPaths);
524
525        return allFiles;
526    }
527
528    /**
529     * Retrieves all files from the defined paths.
530     *
531     * @return a list of files defined via paths.
532     */
533    private List<File> scanPaths() {
534        final List<File> allFiles = new ArrayList<>();
535
536        for (int i = 0; i < paths.size(); i++) {
537            final Path currentPath = paths.get(i);
538            final List<File> pathFiles = scanPath(currentPath, i + 1);
539            allFiles.addAll(pathFiles);
540        }
541
542        return allFiles;
543    }

545    /**
546     * Scans the given path and retrieves all files for the given path.
547     *
548     * @param path      A path to scan.
549     * @param pathIndex The index of the given path. Used in log messages only.
550     * @return A list of files, extracted from the given path.
551     */
552    private List<File> scanPath(Path path, int pathIndex) {
553        final String[] resources = path.list();
554        log(pathIndex + ") Scanning path " + path, Project.MSG_VERBOSE);
555        final List<File> allFiles = new ArrayList<>();
556        int concreteFilesCount = 0;
557
558        for (String resource : resources) {
559            final File file = new File(resource);
560            if (file.isFile()) {
561                concreteFilesCount++;
562                allFiles.add(file);
563            }
564            else {
565                final DirectoryScanner scanner = new DirectoryScanner();
566                scanner.setBasedir(file);
567                scanner.scan();
568                final List<File> scannedFiles = retrieveAllScannedFiles(scanner, pathIndex);
569                allFiles.addAll(scannedFiles);
570            }
571        }
572
573        if (concreteFilesCount > 0) {
574            log(String.format(Locale.ROOT, "%d) Adding %d files from path %s",
575                pathIndex, concreteFilesCount, path), Project.MSG_VERBOSE);
576        }
577
578        return allFiles;
579    }
580
581    /**
582     * Returns the list of files (full path name) to process.
583     *
584     * @return the list of files included via the filesets.
585     */
586    protected List<File> scanFileSets() {
587        final List<File> allFiles = new ArrayList<>();
588
589        for (int i = 0; i < fileSets.size(); i++) {
590            final FileSet fileSet = fileSets.get(i);
591            final DirectoryScanner scanner = fileSet.getDirectoryScanner(getProject());
592            final List<File> scannedFiles = retrieveAllScannedFiles(scanner, i);
593            allFiles.addAll(scannedFiles);
594        }
595
596        return allFiles;
597    }
598
599    /**
600     * Retrieves all matched files from the given scanner.
601     *
602     * @param scanner  A directory scanner. Note, that {@link DirectoryScanner#scan()}
603     *                 must be called before calling this method.
604     * @param logIndex A log entry index. Used only for log messages.
605     * @return A list of files, retrieved from the given scanner.
606     */
607    private List<File> retrieveAllScannedFiles(DirectoryScanner scanner, int logIndex) {
608        final String[] fileNames = scanner.getIncludedFiles();
609        log(String.format(Locale.ROOT, "%d) Adding %d files from directory %s",
610            logIndex, fileNames.length, scanner.getBasedir()), Project.MSG_VERBOSE);
611
612        return Arrays.stream(fileNames)
613            .map(name -> scanner.getBasedir() + File.separator + name)
614            .map(File::new)
615            .collect(Collectors.toList());
616    }

633    /**
634     * Details about a formatter to be used.
635     */
636    public static class Formatter {
637
638        /** The formatter type. */
639        private FormatterType type;
640        /** The file to output to. */
641        private File toFile;
642        /** Whether or not to write to the named file. */
643        private boolean useFile = true;
644
645        /**
646         * Set the type of the formatter.
647         *
648         * @param type the type
649         */
650        public void setType(FormatterType type) {
651            this.type = type;
652        }
653
654        /**
655         * Set the file to output to.
656         *
657         * @param destination destination the file to output to
658         */
659        public void setTofile(File destination) {
660            toFile = destination;
661        }
662
663        /**
664         * Sets whether or not we write to a file if it is provided.
665         *
666         * @param use whether not to use provided file.
667         */
668        public void setUseFile(boolean use) {
669            useFile = use;
670        }
671
672        /**
673         * Creates a listener for the formatter.
674         *
675         * @param task the task running
676         * @return a listener
677         * @throws IOException if an error occurs
678         */
679        public AuditListener createListener(Task task) throws IOException {
680            final AuditListener listener;
681            if (type != null
682                    && E_XML.equals(type.getValue())) {
683                listener = createXmlLogger(task);
684            }
685            else if (type != null
686                    && E_SARIF.equals(type.getValue())) {
687                listener = createSarifLogger(task);
688            }
689            else {
690                listener = createDefaultLogger(task);
691            }
692            return listener;
693        }
694
695        /**
696         * Creates Sarif logger.
697         *
698         * @param task the task to possibly log to
699         * @return an SarifLogger instance
700         * @throws IOException if an error occurs
701         */
702        private AuditListener createSarifLogger(Task task) throws IOException {
703            final AuditListener sarifLogger;
704            if (toFile == null || !useFile) {
705                sarifLogger = new SarifLogger(new LogOutputStream(task, Project.MSG_INFO),
706                        AutomaticBean.OutputStreamOptions.CLOSE);
707            }
708            else {
709                sarifLogger = new SarifLogger(Files.newOutputStream(toFile.toPath()),
710                        AutomaticBean.OutputStreamOptions.CLOSE);
711            }
712            return sarifLogger;
713        }
714
715        /**
716         * Creates default logger.
717         *
718         * @param task the task to possibly log to
719         * @return a DefaultLogger instance
720         * @throws IOException if an error occurs
721         */
722        private AuditListener createDefaultLogger(Task task)
723                throws IOException {
724            final AuditListener defaultLogger;
725            if (toFile == null || !useFile) {
726                defaultLogger = new DefaultLogger(
727                    new LogOutputStream(task, Project.MSG_DEBUG),
728                        AutomaticBean.OutputStreamOptions.CLOSE,
729                        new LogOutputStream(task, Project.MSG_ERR),
730                        AutomaticBean.OutputStreamOptions.CLOSE
731                );
732            }
733            else {
734                final OutputStream infoStream = Files.newOutputStream(toFile.toPath());
735                defaultLogger =
736                        new DefaultLogger(infoStream, AutomaticBean.OutputStreamOptions.CLOSE,
737                                infoStream, AutomaticBean.OutputStreamOptions.NONE);
738            }
739            return defaultLogger;
740        }
741
742        /**
743         * Creates XML logger.
744         *
745         * @param task the task to possibly log to
746         * @return an XMLLogger instance
747         * @throws IOException if an error occurs
748         */
749        private AuditListener createXmlLogger(Task task) throws IOException {
750            final AuditListener xmlLogger;
751            if (toFile == null || !useFile) {
752                xmlLogger = new XMLLogger(new LogOutputStream(task, Project.MSG_INFO),
753                        AutomaticBean.OutputStreamOptions.CLOSE);
754            }
755            else {
756                xmlLogger = new XMLLogger(Files.newOutputStream(toFile.toPath()),
757                        AutomaticBean.OutputStreamOptions.CLOSE);
758            }
759            return xmlLogger;
760        }
761
762    }
