150     *

152     * @param omitIgnoredModules {@code true} if ignored modules should be

158    private ConfigurationLoader(final PropertyResolver overrideProps,
159                                final boolean omitIgnoredModules,

165        this.threadModeSettings = threadModeSettings;

293    /**
294     * Returns the module configurations from a specified input source.
295     * Note that if the source does wrap an open byte or character
296     * stream, clients are required to close that stream by themselves
297     *
298     * @param configSource the input stream to the Checkstyle configuration
299     * @param overridePropsResolver overriding properties
300     * @param ignoredModulesOptions {@code OMIT} if modules with severity
301     *            'ignore' should be omitted, {@code EXECUTE} otherwise
302     * @param threadModeSettings the thread mode configuration
303     * @return the check configurations
304     * @throws CheckstyleException if an error occurs
305     * @noinspection WeakerAccess
306     * @noinspectionreason WeakerAccess - we avoid 'protected' when possible
307     */
308    public static Configuration loadConfiguration(InputSource configSource,
309                                                  PropertyResolver overridePropsResolver,
310                                                  IgnoredModulesOptions ignoredModulesOptions,
311                                                  ThreadModeSettings threadModeSettings)
312            throws CheckstyleException {
313        try {
314            final boolean omitIgnoreModules = ignoredModulesOptions == IgnoredModulesOptions.OMIT;
315            final ConfigurationLoader loader =
316                    new ConfigurationLoader(overridePropsResolver,
317                            omitIgnoreModules, threadModeSettings);
318            loader.parseInputSource(configSource);
319            return loader.configuration;
320        }
321        catch (final SAXParseException ex) {
322            final String message = String.format(Locale.ROOT, SAX_PARSE_EXCEPTION_FORMAT,
323                    UNABLE_TO_PARSE_EXCEPTION_PREFIX,
324                    ex.getMessage(), ex.getLineNumber(), ex.getColumnNumber());
325            throw new CheckstyleException(message, ex);
326        }
327        catch (final ParserConfigurationException | IOException | SAXException ex) {
328            throw new CheckstyleException(UNABLE_TO_PARSE_EXCEPTION_PREFIX, ex);
329        }
330    }

332    /**
333     * Replaces {@code ${xxx}} style constructions in the given value
334     * with the string value of the corresponding data types. This method must remain
335     * outside inner class for easier testing since inner class requires an instance.
336     *
337     * <p>Code copied from
338     * <a href="https://github.com/apache/ant/blob/master/src/main/org/apache/tools/ant/ProjectHelper.java">
339     * ant
340     * </a>
341     *
342     * @param value The string to be scanned for property references. Must
343     *              not be {@code null}.
344     * @param props Mapping (String to String) of property names to their
345     *              values. Must not be {@code null}.
346     * @param defaultValue default to use if one of the properties in value
347     *              cannot be resolved from props.
348     *
349     * @return the original string with the properties replaced.
350     * @throws CheckstyleException if the string contains an opening
351     *                           {@code ${} without a closing
352     *                           {@code }}
353     */
354    private static String replaceProperties(
355            String value, PropertyResolver props, String defaultValue)
356            throws CheckstyleException {
357
358        final List<String> fragments = new ArrayList<>();
359        final List<String> propertyRefs = new ArrayList<>();
360        parsePropertyString(value, fragments, propertyRefs);
361
362        final StringBuilder sb = new StringBuilder(256);
363        final Iterator<String> fragmentsIterator = fragments.iterator();
364        final Iterator<String> propertyRefsIterator = propertyRefs.iterator();
365        while (fragmentsIterator.hasNext()) {
366            String fragment = fragmentsIterator.next();
367            if (fragment == null) {
368                final String propertyName = propertyRefsIterator.next();
369                fragment = props.resolve(propertyName);
370                if (fragment == null) {
371                    if (defaultValue != null) {
372                        sb.replace(0, sb.length(), defaultValue);
373                        break;
374                    }
375                    throw new CheckstyleException(
376                        "Property ${" + propertyName + "} has not been set");
377                }
378            }
379            sb.append(fragment);
380        }
381
382        return sb.toString();
383    }

385    /**
386     * Parses a string containing {@code ${xxx}} style property
387     * references into two collections. The first one is a collection
388     * of text fragments, while the other is a set of string property names.
389     * {@code null} entries in the first collection indicate a property
390     * reference from the second collection.
391     *
392     * <p>Code copied from
393     * <a href="https://github.com/apache/ant/blob/master/src/main/org/apache/tools/ant/ProjectHelper.java">
394     * ant
395     * </a>
396     *
397     * @param value     Text to parse. Must not be {@code null}.
398     * @param fragments Collection to add text fragments to.
399     *                  Must not be {@code null}.
400     * @param propertyRefs Collection to add property names to.
401     *                     Must not be {@code null}.
402     *
403     * @throws CheckstyleException if the string contains an opening
404     *                           {@code ${} without a closing
405     *                           {@code }}
406     */
407    private static void parsePropertyString(String value,
408                                           Collection<String> fragments,
409                                           Collection<String> propertyRefs)
410            throws CheckstyleException {
411        int prev = 0;
412        // search for the next instance of $ from the 'prev' position
413        int pos = value.indexOf(DOLLAR_SIGN, prev);
414        while (pos >= 0) {
415            // if there was any text before this, add it as a fragment
416            if (pos > 0) {
417                fragments.add(value.substring(prev, pos));
418            }
419            // if we are at the end of the string, we tack on a $
420            // then move past it
421            if (pos == value.length() - 1) {
422                fragments.add(DOLLAR_SIGN_STRING);
423                prev = pos + 1;
424            }
425            else if (value.charAt(pos + 1) == '{') {
426                // property found, extract its name or bail on a typo
427                final int endName = value.indexOf('}', pos);
428                if (endName == -1) {
429                    throw new CheckstyleException("Syntax error in property: "
430                                                    + value);
431                }
432                final String propertyName = value.substring(pos + 2, endName);
433                fragments.add(null);
434                propertyRefs.add(propertyName);
435                prev = endName + 1;
436            }
437            else {
438                if (value.charAt(pos + 1) == DOLLAR_SIGN) {
439                    // backwards compatibility two $ map to one mode
440                    fragments.add(DOLLAR_SIGN_STRING);
441                }
442                else {
443                    // new behaviour: $X maps to $X for all values of X!='$'
444                    fragments.add(value.substring(pos, pos + 2));
445                }
446                prev = pos + 2;
447            }
448
449            // search for the next instance of $ from the 'prev' position
450            pos = value.indexOf(DOLLAR_SIGN, prev);
451        }
452        // no more $ signs found
453        // if there is any tail to the file, append it
454        if (prev < value.length()) {
455            fragments.add(value.substring(prev));
456        }
457    }

459    /**
460     * Implements the SAX document handler interfaces, so they do not
461     * appear in the public API of the ConfigurationLoader.
462     */
463    private final class InternalLoader
464        extends XmlLoader {
465
466        /** Module elements. */
467        private static final String MODULE = "module";
468        /** Name attribute. */
469        private static final String NAME = "name";
470        /** Property element. */
471        private static final String PROPERTY = "property";
472        /** Value attribute. */
473        private static final String VALUE = "value";
474        /** Default attribute. */
475        private static final String DEFAULT = "default";
476        /** Name of the severity property. */
477        private static final String SEVERITY = "severity";
478        /** Name of the message element. */
479        private static final String MESSAGE = "message";
480        /** Name of the message element. */
481        private static final String METADATA = "metadata";
482        /** Name of the key attribute. */
483        private static final String KEY = "key";
484
485        /**
486         * Creates a new InternalLoader.
487         *
488         * @throws SAXException if an error occurs
489         * @throws ParserConfigurationException if an error occurs
490         */
491        /* package */ InternalLoader()
492                throws SAXException, ParserConfigurationException {
493            super(createIdToResourceNameMap());
494        }
495
496        @Override
497        public void startElement(String uri,
498                                 String localName,
499                                 String qName,
500                                 Attributes attributes)
501                throws SAXException {
502            if (MODULE.equals(qName)) {
503                // create configuration
504                final String originalName = attributes.getValue(NAME);
505                final String name = threadModeSettings.resolveName(originalName);
506                final DefaultConfiguration conf =
507                    new DefaultConfiguration(name, threadModeSettings);
508
509                if (configuration == null) {
510                    configuration = conf;
511                }
512
513                // add configuration to it's parent
514                if (!configStack.isEmpty()) {
515                    final DefaultConfiguration top =
516                        configStack.peek();
517                    top.addChild(conf);
518                }
519
520                configStack.push(conf);
521            }
522            else if (PROPERTY.equals(qName)) {
523                // extract value and name
524                final String attributesValue = attributes.getValue(VALUE);
525
526                final String value;
527                try {
528                    value = replaceProperties(attributesValue,
529                        overridePropsResolver, attributes.getValue(DEFAULT));
530                }
531                catch (final CheckstyleException ex) {
532                    // -@cs[IllegalInstantiation] SAXException is in the overridden
533                    // method signature
534                    throw new SAXException(ex);
535                }
536
537                final String name = attributes.getValue(NAME);
538
539                // add to attributes of configuration
540                final DefaultConfiguration top =
541                    configStack.peek();
542                top.addProperty(name, value);
543            }
544            else if (MESSAGE.equals(qName)) {
545                // extract key and value
546                final String key = attributes.getValue(KEY);
547                final String value = attributes.getValue(VALUE);
548
549                // add to messages of configuration
550                final DefaultConfiguration top = configStack.peek();
551                top.addMessage(key, value);
552            }
553            else {
554                if (!METADATA.equals(qName)) {
555                    throw new IllegalStateException("Unknown name:" + qName + ".");
556                }
557            }
558        }
559
560        @Override
561        public void endElement(String uri,
562                               String localName,
563                               String qName) throws SAXException {
564            if (MODULE.equals(qName)) {
565                final Configuration recentModule =
566                    configStack.pop();
567
568                // get severity attribute if it exists
569                SeverityLevel level = null;
570                if (containsAttribute(recentModule, SEVERITY)) {
571                    try {
572                        final String severity = recentModule.getProperty(SEVERITY);
573                        level = SeverityLevel.getInstance(severity);
574                    }
575                    catch (final CheckstyleException ex) {
576                        // -@cs[IllegalInstantiation] SAXException is in the overridden
577                        // method signature
578                        throw new SAXException(
579                                "Problem during accessing '" + SEVERITY + "' attribute for "
580                                        + recentModule.getName(), ex);
581                    }
582                }
583
584                // omit this module if these should be omitted and the module
585                // has the severity 'ignore'
586                final boolean omitModule = omitIgnoredModules
587                    && level == SeverityLevel.IGNORE;
588
589                if (omitModule && !configStack.isEmpty()) {
590                    final DefaultConfiguration parentModule =
591                        configStack.peek();
592                    parentModule.removeChild(recentModule);
593                }
594            }
595        }
596
597        /**
598         * Util method to recheck attribute in module.
599         *
600         * @param module module to check
601         * @param attributeName name of attribute in module to find
602         * @return true if attribute is present in module
603         */
604        private boolean containsAttribute(Configuration module, String attributeName) {
605            final String[] names = module.getPropertyNames();
606            final Optional<String> result = Arrays.stream(names)
607                    .filter(name -> name.equals(attributeName)).findFirst();
608            return result.isPresent();
609        }
