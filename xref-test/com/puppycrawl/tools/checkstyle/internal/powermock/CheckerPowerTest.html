51      @Test
52      public void testCatchErrorInProcessFilesMethod() throws Exception {
53          // The idea of the test is to satisfy coverage rate.
54          // An Error indicates serious problems that a reasonable application should not try to
55          // catch, but due to issue https://github.com/checkstyle/checkstyle/issues/2285
56          // we catch errors in 'processFiles' method. Most such errors are abnormal conditions,
57          // that is why we use PowerMockito to reproduce them.
58          final File mock = PowerMockito.mock(File.class);
59          // Assume that I/O error is happened when we try to invoke 'lastModified()' method.
60          final String errorMessage = "Java Virtual Machine is broken"
61              + " or has run out of resources necessary for it to continue operating.";
62          final Error expectedError = new IOError(new InternalError(errorMessage));
63          when(mock.lastModified()).thenThrow(expectedError);
64          when(mock.getAbsolutePath()).thenReturn("testFile");
65          final Checker checker = new Checker();
66          final List<File> filesToProcess = new ArrayList<>();
67          filesToProcess.add(mock);
68          try {
69              checker.process(filesToProcess);
70              fail("IOError is expected!");
71          }
72          // -@cs[IllegalCatchExtended] Testing for catch Error is part of 100% coverage.
73          catch (Error error) {
74              assertThat("Error cause differs from IOError",
75                      error.getCause(), instanceOf(IOError.class));
76              assertThat("Error cause is not InternalError",
77                      error.getCause().getCause(), instanceOf(InternalError.class));
78              assertEquals("Error message is not expected",
79                      errorMessage, error.getCause().getCause().getMessage());
80          }
81      }

114     @Test
115     public void testCatchErrorWithCache() throws Exception {
116         final File cacheFile = temporaryFolder.newFile();
117 
118         final DefaultConfiguration checkerConfig = new DefaultConfiguration("configuration");
119         checkerConfig.addAttribute("charset", StandardCharsets.UTF_8.name());
120         checkerConfig.addAttribute("cacheFile", cacheFile.getPath());
121 
122         final File mock = PowerMockito.mock(File.class);
123         final String errorMessage = "Java Virtual Machine is broken"
124             + " or has run out of resources necessary for it to continue operating.";
125         final Error expectedError = new IOError(new InternalError(errorMessage));
126         when(mock.getAbsolutePath()).thenReturn("testFile");
127         when(mock.getAbsoluteFile()).thenThrow(expectedError);
128         final Checker checker = new Checker();
129         checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
130         checker.configure(checkerConfig);
131         final List<File> filesToProcess = new ArrayList<>();
132         filesToProcess.add(mock);
133         try {
134             checker.process(filesToProcess);
135             fail("IOError is expected!");
136         }
137         // -@cs[IllegalCatchExtended] Testing for catch Error is part of 100% coverage.
138         catch (Error error) {
139             assertThat("Error cause differs from IOError",
140                     error.getCause(), instanceOf(IOError.class));
141             assertEquals("Error message is not expected",
142                     errorMessage, error.getCause().getCause().getMessage());
143 
144             // destroy is called by Main
145             checker.destroy();
146 
147             final Properties cache = new Properties();
148             cache.load(Files.newBufferedReader(cacheFile.toPath()));
149 
150             assertEquals("Cache has unexpected size",
151                     1, cache.size());
152             assertNull("testFile is not in cache",
153                     cache.getProperty("testFile"));
154         }
155     }
