289     /**
290      * Checks that a given field is 'public static final int'.
291      *
292      * @param field field to verify type and visibility of
293      * @return true if field is declared as 'public static final int'
294      */
295     private static boolean isPublicStaticFinalInt(Field field) {
296         final Class<?> fieldType = field.getType();
297         final int mods = field.getModifiers();
298         return fieldType.equals(Integer.TYPE)
299                 && Modifier.isPublic(mods)
300                 && Modifier.isStatic(mods)
301                 && Modifier.isFinal(mods);
302     }
