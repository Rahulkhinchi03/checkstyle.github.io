157     @Override
158     public void beginJavadocTree(DetailNode rootAst) {
159         if (isTopLevelClassJavadoc()) {
160             moduleDetails = new ModuleDetails();
161             toScan = false;
162             scrapingViolationMessageList = false;
163             propertySectionStartIdx = -1;
164             exampleSectionStartIdx = -1;
165             parentSectionStartIdx = -1;
166 
167             final String filePath = getFileContents().getFileName();
168             String moduleName = getModuleSimpleName();
169             final String checkModuleExtension = "Check";
170             if (moduleName.contains(checkModuleExtension)) {
171                 moduleName = moduleName.substring(0, moduleName.indexOf(checkModuleExtension));
172             }
173             moduleDetails.setName(moduleName);
174             moduleDetails.setFullQualifiedName(getPackageName(filePath));
175             moduleDetails.setModuleType(getModuleType());
176         }
177     }

179     @Override
180     public void visitJavadocToken(DetailNode ast) {
181         if (toScan) {
182             scrapeContent(ast);
183         }
184 
185         if (ast.getType() == JavadocTokenTypes.JAVADOC) {
186             final DetailAST parent = getParent(getBlockCommentAst());
187             if (parent != null && parent.getType() == TokenTypes.CLASS_DEF) {
188                 rootNode = ast;
189                 toScan = true;
190             }
191         }
192         else if (ast.getType() == JavadocTokenTypes.SINCE_LITERAL) {
193             toScan = false;
194         }
195     }

197     @Override
198     public void finishJavadocTree(DetailNode rootAst) {
199         moduleDetails.setDescription(getDescriptionText());
200         if (isTopLevelClassJavadoc()) {
201             if (getFileContents().getFileName().contains("test")) {
202                 MODULE_DETAILS_STORE.put(moduleDetails.getFullQualifiedName(), moduleDetails);
203             }
204             else {
205                 try {
206                     XmlMetaWriter.write(moduleDetails);
207                 }
208                 catch (TransformerException | ParserConfigurationException ex) {
209                     throw new IllegalStateException("Failed to write metadata into XML file for "
210                             + "module: " + getModuleSimpleName(), ex);
211                 }
212             }
213         }
214     }

216     /**
217      * Method containing the core logic of scraping. This keeps track and decides which phase of
218      * scraping we are in, and accordingly call other subroutines.
219      *
220      * @param ast javadoc ast
221      */
222     public void scrapeContent(DetailNode ast) {
223         if (ast.getType() == JavadocTokenTypes.PARAGRAPH) {
224             if (isParentText(ast)) {
225                 parentSectionStartIdx = getParentIndexOf(ast);
226                 moduleDetails.setParent(getParentText(ast));
227             }
228             else if (isViolationMessagesText(ast)) {
229                 scrapingViolationMessageList = true;
230             }
231             else if (exampleSectionStartIdx == -1
232                     && isExamplesText(ast)) {
233                 exampleSectionStartIdx = getParentIndexOf(ast);
234             }
235         }
236         else if (ast.getType() == JavadocTokenTypes.LI) {
237             if (isPropertyList(ast)) {
238                 if (propertySectionStartIdx == -1) {
239                     propertySectionStartIdx = getParentIndexOf(ast);
240                 }
241                 moduleDetails.addToProperties(createProperties(ast));
242             }
243             else if (scrapingViolationMessageList) {
244                 moduleDetails.addToViolationMessages(getViolationMessages(ast));
245             }
246         }
247     }

315     /**
316      * Clean up the default token text by removing hyperlinks, and only keeping token type text.
317      *
318      * @param initialText unclean text
319      * @return clean text
320      */
321     private static String cleanDefaultTokensText(String initialText) {
322         final Set<String> tokens = new LinkedHashSet<>();
323         final Matcher matcher = TOKEN_TEXT_PATTERN.matcher(initialText);
324         while (matcher.find()) {
325             tokens.add(matcher.group(0));
326         }
327         return String.join(",", tokens);
328     }

330     /**
331      * Performs a DFS of the subtree with a node as the root and constructs the text of that
332      * tree, ignoring JavadocToken texts.
333      *
334      * @param node root node of subtree
335      * @param childLeftLimit the left index of root children from where to scan
336      * @param childRightLimit the right index of root children till where to scan
337      * @return constructed text of subtree
338      */
339     private static String constructSubTreeText(DetailNode node, int childLeftLimit,
340                                                int childRightLimit) {
341         final StringBuilder result = new StringBuilder(1024);
342         DetailNode detailNode = node;
343 
344         final Deque<DetailNode> stack = new ArrayDeque<>();
345         stack.addFirst(detailNode);
346         final Set<DetailNode> visited = new HashSet<>();
347         while (!stack.isEmpty()) {
348             detailNode = stack.getFirst();
349             stack.removeFirst();
350 
351             if (!visited.contains(detailNode)) {
352                 final String childText = detailNode.getText();
353                 if (detailNode.getType() != JavadocTokenTypes.LEADING_ASTERISK
354                         && !TOKEN_TEXT_PATTERN.matcher(childText).matches()) {
355                     result.insert(0, detailNode.getText());
356                 }
357                 visited.add(detailNode);
358             }
359 
360             for (DetailNode child : detailNode.getChildren()) {
361                 if (child.getParent().equals(node)
362                         && (child.getIndex() < childLeftLimit
363                         || child.getIndex() > childRightLimit)) {
364                     continue;
365                 }
366                 if (!visited.contains(child)) {
367                     stack.addFirst(child);
368                 }
369             }
370         }
371         return result.toString().trim();
372     }

374     /**
375      * Create the description text with starting index as 0 and ending index would be the first
376      * valid non zero index amongst in the order of {@code propertySectionStartIdx},
377      * {@code exampleSectionStartIdx} and {@code parentSectionStartIdx}.
378      *
379      * @return description text
380      */
381     private String getDescriptionText() {
382         final int descriptionEndIdx;
383         if (propertySectionStartIdx > -1) {
384             descriptionEndIdx = propertySectionStartIdx;
385         }
386         else if (exampleSectionStartIdx > -1) {
387             descriptionEndIdx = exampleSectionStartIdx;
388         }
389         else {
390             descriptionEndIdx = parentSectionStartIdx;
391         }
392         return constructSubTreeText(rootNode, 0, descriptionEndIdx - 1);
393     }

395     /**
396      * Create property default text, which is either normal property value or list of tokens.
397      *
398      * @param nodeLi list item javadoc node
399      * @return default property text
400      */
401     private static String getPropertyDefaultText(DetailNode nodeLi) {
402         String result = "";
403         final Optional<DetailNode> defaultValueNodeOpt = getFirstChildOfMatchingText(nodeLi,
404                 DEFAULT_VALUE_TAG);
405         if (defaultValueNodeOpt.isPresent()) {
406             final DetailNode defaultValueNode = defaultValueNodeOpt.get();
407             final Optional<DetailNode> propertyDefaultValueTagNode = getFirstChildOfType(nodeLi,
408                     JavadocTokenTypes.JAVADOC_INLINE_TAG, defaultValueNode.getIndex() + 1);
409             DetailNode propertyDefaultValueTag = null;
410             if (propertyDefaultValueTagNode.isPresent()) {
411                 propertyDefaultValueTag = propertyDefaultValueTagNode.get();
412             }
413             if (propertyDefaultValueTag == null) {
414                 final String tokenText = constructSubTreeText(nodeLi,
415                         defaultValueNode.getIndex(), nodeLi.getChildren().length);
416                 result = cleanDefaultTokensText(tokenText);
417             }
418             else {
419                 result = getTextFromTag(propertyDefaultValueTag);
420             }
421         }
422         return result;
423     }

579     /**
580      * Retrieve package name of module from the absolute file path.
581      *
582      * @param filePath absolute file path
583      * @return package name
584      */
585     private static String getPackageName(String filePath) {
586         final Deque<String> result = new ArrayDeque<>();
587         final String[] filePathTokens = FILE_SEPARATOR_PATTERN.split(filePath);
588         for (int i = filePathTokens.length - 1; i >= 0; i--) {
589             if ("java".equals(filePathTokens[i]) || "resources".equals(filePathTokens[i])) {
590                 break;
591             }
592             result.addFirst(filePathTokens[i]);
593         }
594         final String fileName = result.removeLast();
595         result.addLast(fileName.substring(0, fileName.length() - JAVA_FILE_EXTENSION.length()));
596         return String.join(".", result);
597     }

675                 && PARENT_TAG.matcher(resultNode.get().getText()).matches();
